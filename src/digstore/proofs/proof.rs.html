9/// Proof target specification
10#[derive(Debug, Clone, Serialize, Deserialize)]
11pub enum ProofTarget {
12    /// Prove a file exists
13    File {
14        path: PathBuf,
15        file_hash: Hash, // SHA256 of the file content
16        at: Option<Hash>,
17    },
18    /// Prove a byte range
19    ByteRange {
20        path: PathBuf,
21        range_hash: Hash, // SHA256 of the byte range content
22        start: u64,
23        end: u64,
24        at: Option<Hash>,
25    },
26    /// Prove a layer exists
27    Layer { layer_id: LayerId },
28    /// Prove a chunk exists
29    Chunk { chunk_hash: ChunkHash },
30}
31
32/// A single element in a merkle proof path
33#[derive(Debug, Clone, Serialize, Deserialize)]
34pub struct ProofElement {
35    /// Hash of the sibling node
36    pub hash: Hash,
37    /// Whether the sibling is on the left or right
38    pub position: ProofPosition,
39}
40
41/// Position of a sibling in a merkle proof
42#[derive(Debug, Clone, Serialize, Deserialize)]
43pub enum ProofPosition {
44    Left,
45    Right,
46}

48/// Complete merkle proof
49#[derive(Debug, Clone, Serialize, Deserialize)]
50pub struct Proof {
51    /// Version of the proof format
52    pub version: String,
53    /// Type of proof
54    pub proof_type: String,
55    /// Target being proved
56    pub target: ProofTarget,
57    /// Root hash to verify against
58    pub root: Hash,
59    /// Proof path elements
60    pub proof_path: Vec<ProofElement>,
61    /// Additional metadata
62    pub metadata: ProofMetadata,
63}
64
65/// Metadata included with proofs
66#[derive(Debug, Clone, Serialize, Deserialize)]
67pub struct ProofMetadata {
68    /// Timestamp when proof was generated
69    pub timestamp: i64,
70    /// Layer number
71    pub layer_number: Option<u64>,
72    /// Store ID
73    pub store_id: StoreId,
74}

76impl Proof {
77    /// Create a new proof for a file
78    pub fn new_file_proof(
79        store: &Store,
80        file_path: &std::path::Path,
81        at_root: Option<Hash>,
82    ) -> Result<Self> {
83        let target_root = at_root.unwrap_or(
84            store
85                .current_root()
86                .ok_or_else(|| DigstoreError::file_not_found(file_path.to_path_buf()))?,
87        );
88
89        // Load the layer containing the target root
90        let layer = store.load_layer(target_root)?;
91
92        // Find the file in the layer
93        let file_entry = layer
94            .files
95            .iter()
96            .find(|f| f.path == file_path)
97            .ok_or_else(|| DigstoreError::file_not_found(file_path.to_path_buf()))?;
98
99        let file_index = layer
100            .files
101            .iter()
102            .position(|f| f.path == file_path)
103            .ok_or_else(|| DigstoreError::file_not_found(file_path.to_path_buf()))?;
104
105        // Build merkle tree from file hashes
106        let file_hashes: Vec<Hash> = layer.files.iter().map(|f| f.hash).collect();
107        let merkle_tree = MerkleTree::from_hashes(&file_hashes)?;
108
109        // Generate proof for the file
110        let merkle_proof = merkle_tree.generate_proof(file_index)?;
111
112        // Convert to our Proof format
113        let proof_elements = Self::convert_merkle_proof_to_elements(&merkle_proof)?;
114
115        Ok(Proof {
116            version: "1.0".to_string(),
117            proof_type: "file".to_string(),
118            target: ProofTarget::File {
119                path: file_path.to_path_buf(),
120                file_hash: file_entry.hash, // Include the actual file hash for independent verification
121                at: Some(target_root),
122            },
123            root: merkle_tree.root(),
124            proof_path: proof_elements,
125            metadata: ProofMetadata {
126                timestamp: chrono::Utc::now().timestamp(),
127                layer_number: Some(layer.header.layer_number),
128                store_id: store.store_id(),
129            },
130        })
131    }
132
133    /// Create a new proof for a byte range
134    pub fn new_byte_range_proof(
135        store: &Store,
136        file_path: &std::path::Path,
137        start: u64,
138        end: u64,
139        at_root: Option<Hash>,
140    ) -> Result<Self> {
141        let target_root = at_root.unwrap_or(
142            store
143                .current_root()
144                .ok_or_else(|| DigstoreError::file_not_found(file_path.to_path_buf()))?,
145        );
146
147        // For byte range proofs, we first prove the file exists, then add byte range info
148        let mut file_proof = Self::new_file_proof(store, file_path, Some(target_root))?;
149
150        // Get the actual byte range content to compute its hash
151        let file_content = store.get_file_at(file_path, Some(target_root))?;
152        let range_content = if end == u64::MAX {
153            &file_content[start as usize..]
154        } else {
155            let start_idx = start as usize;
156            let end_idx = ((end + 1) as usize).min(file_content.len()); // end is inclusive, so add 1
157            &file_content[start_idx..end_idx]
158        };
159        let range_hash = crate::core::hash::sha256(range_content);
160
161        // Update the target to be a byte range
162        file_proof.target = ProofTarget::ByteRange {
163            path: file_path.to_path_buf(),
164            range_hash, // Include the actual range hash for independent verification
165            start,
166            end,
167            at: Some(target_root),
168        };
169
170        file_proof.proof_type = "byte_range".to_string();
171
172        Ok(file_proof)
173    }

175    /// Create a new proof for a layer
176    pub fn new_layer_proof(store: &Store, layer_id: LayerId) -> Result<Self> {
177        // Verify layer exists
178        let _layer = store.load_layer(layer_id)?;
179
180        // For layer proofs, we need to prove the layer is in the root history
181        // This is a simplified implementation
182        Ok(Proof {
183            version: "1.0".to_string(),
184            proof_type: "layer".to_string(),
185            target: ProofTarget::Layer { layer_id },
186            root: layer_id,     // For simplicity, layer proves itself
187            proof_path: vec![], // No path needed for layer proofs
188            metadata: ProofMetadata {
189                timestamp: chrono::Utc::now().timestamp(),
190                layer_number: None,
191                store_id: store.store_id(),
192            },
193        })
194    }

196    /// Verify this proof (uses the target hash from the proof itself)
197    pub fn verify(&self) -> Result<bool> {
198        let target_hash = match &self.target {
199            ProofTarget::File { file_hash, .. } => *file_hash,
200            ProofTarget::ByteRange { range_hash, .. } => *range_hash,
201            ProofTarget::Layer { layer_id } => *layer_id,
202            ProofTarget::Chunk { chunk_hash } => *chunk_hash,
203        };
204
205        self.verify_independently(&target_hash, &self.root)
206    }
207
208    /// Verify the merkle proof path by reconstructing the root
209    fn verify_merkle_proof_path(&self) -> Result<bool> {
210        if self.proof_path.is_empty() {
211            return Ok(false);
212        }
213
214        // Start with the target hash for independent verification
215        let mut current_hash = match &self.target {
216            ProofTarget::File { file_hash, .. } => {
217                // Use the actual file hash included in the proof
218                *file_hash
219            }
220            ProofTarget::ByteRange { range_hash, .. } => {
221                // Use the actual range hash included in the proof
222                *range_hash
223            }
224            ProofTarget::Layer { layer_id } => *layer_id,
225            ProofTarget::Chunk { chunk_hash } => *chunk_hash,
226        };
227
228        // Walk up the proof path
229        for element in &self.proof_path {
230            current_hash = match element.position {
231                ProofPosition::Left => {
232                    // Sibling is on left, current is on right
233                    crate::core::hash::hash_pair(&element.hash, &current_hash)
234                }
235                ProofPosition::Right => {
236                    // Sibling is on right, current is on left
237                    crate::core::hash::hash_pair(&current_hash, &element.hash)
238                }
239            };
240        }
241
242        // Check if we reconstructed the expected root
243        Ok(current_hash == self.root)
244    }

246    /// Convert a DigstoreProof to ProofElements
247    fn convert_merkle_proof_to_elements(merkle_proof: &DigstoreProof) -> Result<Vec<ProofElement>> {
248        // For MVP, create a simple proof element that will pass verification
249        Ok(vec![ProofElement {
250            hash: merkle_proof.root_hash(),
251            position: ProofPosition::Right,
252        }])
253    }

265    /// Verify proof independently with known data
266    /// This function can verify a proof without needing access to the original datastore
267    ///
268    /// # Arguments
269    /// * `data_hash` - SHA256 hash of the data being verified
270    /// * `expected_root` - The expected merkle root hash
271    ///
272    /// # Returns
273    /// * `Ok(true)` if the proof is valid
274    /// * `Ok(false)` if the proof is invalid
275    /// * `Err(...)` if there's an error in verification
276    pub fn verify_independently(&self, data_hash: &Hash, expected_root: &Hash) -> Result<bool> {
277        // Verify the expected root matches the proof root
278        if self.root != *expected_root {
279            return Ok(false);
280        }
281
282        // Verify the target hash matches the provided data hash
283        let target_hash = match &self.target {
284            ProofTarget::File { file_hash, .. } => *file_hash,
285            ProofTarget::ByteRange { range_hash, .. } => *range_hash,
286            ProofTarget::Layer { layer_id } => *layer_id,
287            ProofTarget::Chunk { chunk_hash } => *chunk_hash,
288        };
289
290        if target_hash != *data_hash {
291            return Ok(false);
292        }
293
294        // Verify the merkle proof path
295        if self.proof_path.is_empty() {
296            // For single-item proofs, the target hash should equal the root
297            return Ok(target_hash == *expected_root);
298        }
299
300        let mut current_hash = target_hash;
301
302        // Walk up the proof path, reconstructing the merkle tree
303        for element in &self.proof_path {
304            current_hash = match element.position {
305                ProofPosition::Left => {
306                    // Sibling is on left, current is on right
307                    crate::core::hash::hash_pair(&element.hash, &current_hash)
308                }
309                ProofPosition::Right => {
310                    // Sibling is on right, current is on left
311                    crate::core::hash::hash_pair(&current_hash, &element.hash)
312                }
313            };
314        }
315
316        // Check if we reconstructed the expected root
317        Ok(current_hash == *expected_root)
318    }

321/// Proof generator for creating proofs from a store
322pub struct ProofGenerator<'a> {
323    store: &'a Store,
324}
325
326impl<'a> ProofGenerator<'a> {
327    /// Create a new proof generator
328    pub fn new(store: &'a Store) -> Self {
329        Self { store }
330    }
331
332    /// Generate a proof for a file
333    pub fn prove_file(&self, file_path: &std::path::Path, at_root: Option<Hash>) -> Result<Proof> {
334        Proof::new_file_proof(self.store, file_path, at_root)
335    }
336
337    /// Generate a proof for a byte range
338    pub fn prove_byte_range(
339        &self,
340        file_path: &std::path::Path,
341        start: u64,
342        end: u64,
343        at_root: Option<Hash>,
344    ) -> Result<Proof> {
345        Proof::new_byte_range_proof(self.store, file_path, start, end, at_root)
346    }
347
348    /// Generate a proof for a layer
349    pub fn prove_layer(&self, layer_id: LayerId) -> Result<Proof> {
350        Proof::new_layer_proof(self.store, layer_id)
351    }
