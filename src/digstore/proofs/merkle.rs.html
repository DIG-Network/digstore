1//! Merkle tree implementation using rs_merkle
2
3use crate::core::{error::*, types::*};
4use rs_merkle::{Hasher, MerkleProof as RsMerkleProof, MerkleTree as RsMerkleTree};
5use sha2::{Digest, Sha256};
6
7/// Custom hasher implementation for rs_merkle
8#[derive(Clone)]
9pub struct Sha256Hasher;
10
11impl Hasher for Sha256Hasher {
12    type Hash = [u8; 32];
13
14    fn hash(data: &[u8]) -> Self::Hash {
15        let mut hasher = Sha256::new();
16        hasher.update(data);
17        hasher.finalize().into()
18    }
19}
20
21/// Merkle tree wrapper for Digstore Min
22pub struct MerkleTree {
23    /// The underlying rs_merkle tree
24    inner: RsMerkleTree<Sha256Hasher>,
25    /// Root hash of the tree
26    root: Hash,
27    /// Original leaf hashes
28    leaves: Vec<Hash>,
29}
30
31impl MerkleTree {
32    /// Build a merkle tree from a list of hashes
33    pub fn from_hashes(hashes: &[Hash]) -> Result<Self> {
34        if hashes.is_empty() {
35            return Err(DigstoreError::MerkleTreeFailed {
36                reason: "Cannot create merkle tree from empty hash list".to_string(),
37            });
38        }
39
40        // Convert our Hash types to [u8; 32] for rs_merkle
41        let leaves: Vec<[u8; 32]> = hashes.iter().map(|h| *h.as_bytes()).collect();
42
43        // Build the tree
44        let tree = RsMerkleTree::<Sha256Hasher>::from_leaves(&leaves);
45
46        // Get the root hash
47        let root_bytes = tree.root().ok_or_else(|| DigstoreError::MerkleTreeFailed {
48            reason: "Failed to get merkle tree root".to_string(),
49        })?;
50
51        let root = Hash::from_bytes(root_bytes);
52
53        Ok(Self {
54            inner: tree,
55            root,
56            leaves: hashes.to_vec(),
57        })
58    }
59
60    /// Get the root hash
61    pub fn root(&self) -> Hash {
62        self.root
63    }
64
65    /// Get the number of leaves
66    pub fn leaf_count(&self) -> usize {
67        self.leaves.len()
68    }
69
70    /// Generate a proof for a specific leaf index
71    pub fn generate_proof(&self, leaf_index: usize) -> Result<DigstoreProof> {
72        if leaf_index >= self.leaves.len() {
73            return Err(DigstoreError::MerkleTreeFailed {
74                reason: format!(
75                    "Leaf index {} out of bounds (tree has {} leaves)",
76                    leaf_index,
77                    self.leaves.len()
78                ),
79            });
80        }
81
82        let proof = self.inner.proof(&[leaf_index]);
83        let proof_bytes = proof.to_bytes();
84
85        Ok(DigstoreProof {
86            leaf_index,
87            leaf_hash: self.leaves[leaf_index],
88            proof_bytes,
89            root_hash: self.root,
90        })
91    }
92
93    /// Verify a proof against this tree's root
94    pub fn verify_proof(&self, merkle_proof: &DigstoreProof) -> bool {
95        // Parse the proof from bytes
96        if let Ok(proof) =
97            RsMerkleProof::<Sha256Hasher>::try_from(merkle_proof.proof_bytes.as_slice())
98        {
99            // Verify the proof
100            proof.verify(
101                *self.root.as_bytes(),
102                &[merkle_proof.leaf_index],
103                &[*merkle_proof.leaf_hash.as_bytes()],
104                self.leaves.len(),
105            )
106        } else {
107            false
108        }
109    }
110
111    /// Get all leaf hashes
112    pub fn leaves(&self) -> &[Hash] {
113        &self.leaves
114    }
115}

117/// A merkle proof for a specific leaf
118#[derive(Debug, Clone)]
119pub struct DigstoreProof {
120    /// Index of the leaf being proved
121    pub leaf_index: usize,
122    /// Hash of the leaf being proved
123    pub leaf_hash: Hash,
124    /// Serialized proof bytes
125    pub proof_bytes: Vec<u8>,
126    /// Root hash this proof is against
127    pub root_hash: Hash,
128}
129
130impl DigstoreProof {
131    /// Verify this proof against a given root hash
132    pub fn verify(&self, expected_root: &Hash) -> bool {
133        if self.root_hash != *expected_root {
134            return false;
135        }
136
137        if let Ok(proof) = RsMerkleProof::<Sha256Hasher>::try_from(self.proof_bytes.as_slice()) {
138            proof.verify(
139                *expected_root.as_bytes(),
140                &[self.leaf_index],
141                &[*self.leaf_hash.as_bytes()],
142                // Note: We need the leaf count, but it's not stored in the proof
143                // For now, we'll assume verification happens against the original tree
144                usize::MAX, // This is a limitation we'll address
145            )
146        } else {
147            false
148        }
149    }
150
151    /// Get the leaf index
152    pub fn leaf_index(&self) -> usize {
153        self.leaf_index
154    }
155
156    /// Get the leaf hash
157    pub fn leaf_hash(&self) -> Hash {
158        self.leaf_hash
159    }
160
161    /// Get the root hash
162    pub fn root_hash(&self) -> Hash {
163        self.root_hash
164    }
165}

172    #[test]
173    fn test_merkle_tree_creation() -> Result<()> {
174        let hashes = vec![
175            sha256(b"file1"),
176            sha256(b"file2"),
177            sha256(b"file3"),
178            sha256(b"file4"),
179        ];
180
181        let tree = MerkleTree::from_hashes(&hashes)?;
182
183        assert_eq!(tree.leaf_count(), 4);
184        assert_ne!(tree.root(), Hash::zero());
185        assert_eq!(tree.leaves(), &hashes);

187        Ok(())
