1//! .digstore file management

3use crate::core::{error::*, types::*};
4use chrono::Utc;
5use serde::{Deserialize, Serialize};
6use std::path::Path;
7
8/// Contents of a .digstore file
9#[derive(Debug, Clone, Serialize, Deserialize)]
10pub struct DigstoreFile {
11    /// Format version
12    pub version: String,
13    /// Store ID this project links to
14    pub store_id: String,
15    /// Whether encryption is enabled (always false for digstore_min)
16    pub encrypted: bool,
17    /// When the link was created
18    pub created_at: String,
19    /// Last access time
20    pub last_accessed: String,
21    /// Optional repository name
22    pub repository_name: Option<String>,
23}
24
25impl DigstoreFile {
26    /// Create a new .digstore file configuration
27    pub fn new(store_id: StoreId, repository_name: Option<String>) -> Self {
28        let now = Utc::now().to_rfc3339();
29        Self {
30            version: "1.0.0".to_string(),
31            store_id: store_id.to_hex(),
32            encrypted: false,
33            created_at: now.clone(),
34            last_accessed: now,
35            repository_name,
36        }
37    }
38
39    /// Load .digstore file from disk
40    pub fn load(path: &Path) -> Result<Self> {
41        let content = std::fs::read_to_string(path).map_err(|e| DigstoreError::Io(e))?;
42
43        let digstore_file: DigstoreFile =
44            toml::from_str(&content).map_err(|e| DigstoreError::ConfigurationError {
45                reason: format!("Failed to parse .digstore file: {}", e),
46            })?;
47
48        // Validate the format
49        if digstore_file.version != "1.0.0" {
50            return Err(DigstoreError::ConfigurationError {
51                reason: format!("Unsupported .digstore version: {}", digstore_file.version),
52            });
53        }
54
55        // Validate store ID format
56        if digstore_file.store_id.len() != 64 {
57            return Err(DigstoreError::invalid_store_id(format!(
58                "Store ID must be 64 hex characters, got {}",
59                digstore_file.store_id.len()
60            )));
61        }
62
63        Ok(digstore_file)
64    }
65
66    /// Save .digstore file to disk
67    pub fn save(&self, path: &Path) -> Result<()> {
68        let content =
69            toml::to_string_pretty(self).map_err(|e| DigstoreError::ConfigurationError {
70                reason: format!("Failed to serialize .digstore file: {}", e),
71            })?;
72
73        std::fs::write(path, content).map_err(|e| DigstoreError::Io(e))?;
74
75        Ok(())
76    }
77
78    /// Update last accessed time
79    pub fn update_last_accessed(&mut self) {
80        self.last_accessed = Utc::now().to_rfc3339();
81    }
82
83    /// Get the store ID as a Hash
84    pub fn get_store_id(&self) -> Result<StoreId> {
85        Hash::from_hex(&self.store_id).map_err(|_| {
86            DigstoreError::invalid_store_id(format!(
87                "Invalid store ID in .digstore file: {}",
88                self.store_id
89            ))
90        })
91    }
92
93    /// Check if this .digstore file is valid
94    pub fn is_valid(&self) -> bool {
95        self.version == "1.0.0" 
96            && self.store_id.len() == 64
97            && !self.encrypted // digstore_min doesn't support encryption
98            && Hash::from_hex(&self.store_id).is_ok()
99    }
100}

108    #[test]
109    fn test_digstore_file_creation() {
110        let store_id =
111            Hash::from_hex("a3f5c8d9e2b1f4a6c9d8e7f2a5b8c1d4e7f0a3b6c9d2e5f8b1c4d7e0a3b6c9d2")
112                .unwrap();
113        let digstore_file = DigstoreFile::new(store_id, Some("test-repo".to_string()));
114
115        assert_eq!(digstore_file.version, "1.0.0");
116        assert_eq!(
117            digstore_file.store_id,
118            "a3f5c8d9e2b1f4a6c9d8e7f2a5b8c1d4e7f0a3b6c9d2e5f8b1c4d7e0a3b6c9d2"
119        );
120        assert_eq!(digstore_file.encrypted, false);
121        assert_eq!(digstore_file.repository_name, Some("test-repo".to_string()));
122        assert!(digstore_file.is_valid());
123    }

125    #[test]
126    fn test_digstore_file_roundtrip() -> Result<()> {
127        let store_id =
128            Hash::from_hex("a3f5c8d9e2b1f4a6c9d8e7f2a5b8c1d4e7f0a3b6c9d2e5f8b1c4d7e0a3b6c9d2")
129                .unwrap();
130        let original = DigstoreFile::new(store_id, Some("test-repo".to_string()));
131
132        let mut temp_file = NamedTempFile::new().unwrap();
133        original.save(temp_file.path())?;
134
135        let loaded = DigstoreFile::load(temp_file.path())?;
136
137        assert_eq!(original.version, loaded.version);
138        assert_eq!(original.store_id, loaded.store_id);
139        assert_eq!(original.encrypted, loaded.encrypted);
140        assert_eq!(original.repository_name, loaded.repository_name);
141
142        Ok(())
143    }
