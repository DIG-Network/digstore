50/// Execute the layers command
51pub fn execute(
52    layer_hash: Option<String>,
53    json: bool,
54    list: bool,
55    size: bool,
56    files: bool,
57    chunks: bool,
58) -> Result<()> {
59    let args = LayersArgs {
60        layer_hash,
61        json,
62        list,
63        size,
64        files,
65        chunks,
66    };
67
68    let current_dir = std::env::current_dir()?;
69    let store = Store::open(&current_dir)?;
70
71    if let Some(hash_str) = &args.layer_hash {
72        // Analyze specific layer
73        let layer_hash = crate::core::types::Hash::from_hex(hash_str)
74            .map_err(|_| anyhow::anyhow!("Invalid layer hash: {}", hash_str))?;
75
76        analyze_specific_layer(&store, layer_hash, &args)?;
77    } else if args.list {
78        // List all layers
79        list_all_layers(&store, &args)?;
80    } else {
81        // Show current layer
82        if let Some(current_root) = store.current_root() {
83            analyze_specific_layer(&store, current_root, &args)?;
84        } else {
85            if args.json {
86                println!("{}", json!({"error": "No current layer found"}));
87            } else {
88                println!("{}", "No current layer found".yellow());
89                println!("  {} Use 'digstore commit' to create a layer", "→".cyan());
90            }
91        }
92    }
93
94    Ok(())
95}

97fn analyze_specific_layer(
98    store: &Store,
99    layer_hash: crate::core::types::Hash,
100    args: &LayersArgs,
101) -> Result<()> {
102    let layer = store.load_layer(layer_hash)?;
103
104    // Get layer file size from archive
105    let file_size = if let Some(entry) = store
106        .archive
107        .list_layers()
108        .iter()
109        .find(|(hash, _)| *hash == layer_hash)
110    {
111        entry.1.size
112    } else {
113        0
114    };
115
116    let layer_info = LayerInfo {
117        hash: layer_hash.to_hex(),
118        layer_type: format!(
119            "{:?}",
120            layer
121                .header
122                .get_layer_type()
123                .unwrap_or(crate::core::types::LayerType::Full)
124        ),
125        generation: layer.header.layer_number,
126        parent_hash: layer.header.get_parent_hash().to_hex(),
127        timestamp: layer.header.timestamp as i64,
128        file_size,
129        files_count: layer.files.len(),
130        chunks_count: layer.chunks.len(),
131        total_file_size: layer.files.iter().map(|f| f.size).sum(),
132        commit_message: layer.metadata.message.clone(),
133        author: layer.metadata.author.clone(),
134    };
135
136    if args.json {
137        show_layer_json(&layer_info, &layer, args)?;
138    } else {
139        show_layer_human(&layer_info, &layer, args)?;
140    }
141
142    Ok(())
143}

145fn list_all_layers(store: &Store, args: &LayersArgs) -> Result<()> {
146    let mut layers = Vec::new();
147
148    // Get all layers from archive
149    let archive_layers = store.archive.list_layers();
150
151    for (layer_hash, entry) in archive_layers {
152        if layer_hash == crate::core::types::Hash::zero() {
153            // Special handling for Layer 0 (metadata layer)
154            layers.push(LayerInfo {
155                hash: layer_hash.to_hex(),
156                layer_type: "Metadata".to_string(),
157                generation: 0,
158                parent_hash: Hash::zero().to_hex(),
159                timestamp: chrono::Utc::now().timestamp(),
160                file_size: entry.size,
161                files_count: 0,
162                chunks_count: 0,
163                total_file_size: 0,
164                commit_message: Some("Repository metadata".to_string()),
165                author: Some("System".to_string()),
166            });
167        } else {
168            // Load regular layer to get detailed information
169            if let Ok(layer) = store.load_layer(layer_hash) {
170                layers.push(LayerInfo {
171                    hash: layer_hash.to_hex(),
172                    layer_type: format!(
173                        "{:?}",
174                        layer
175                            .header
176                            .get_layer_type()
177                            .unwrap_or(crate::core::types::LayerType::Full)
178                    ),
179                    generation: layer.header.layer_number,
180                    parent_hash: layer.header.get_parent_hash().to_hex(),
181                    timestamp: layer.header.timestamp as i64,
182                    file_size: entry.size,
183                    files_count: layer.files.len(),
184                    chunks_count: layer.chunks.len(),
185                    total_file_size: layer.files.iter().map(|f| f.size).sum(),
186                    commit_message: layer.metadata.message.clone(),
187                    author: layer.metadata.author.clone(),
188                });
189            }
190        }
191    }
192
193    // Sort by generation
194    layers.sort_by_key(|l| l.generation);
195
196    if args.json {
197        println!(
198            "{}",
199            serde_json::to_string_pretty(&json!({
200                "layers": layers,
201                "total_layers": layers.len()
202            }))?
203        );
204    } else {
205        println!("{}", "Layer List".green().bold());
206        println!("{}", "═".repeat(50).green());
207
208        if layers.is_empty() {
209            println!("{}", "No layers found".yellow());
210            return Ok(());
211        }
212
213        for layer in &layers {
214            println!("\n{} {}", "Layer".bold(), layer.hash.cyan());
215            println!("  Type: {}", layer.layer_type);
216            println!("  Generation: {}", layer.generation);
217            println!("  Size: {}", format_bytes(layer.file_size));
218            println!("  Files: {}", layer.files_count);
219
220            if let Some(message) = &layer.commit_message {
221                println!("  Message: {}", message.bright_white());
222            }
223        }
224
225        println!("\n{}", format!("Total layers: {}", layers.len()).cyan());
226    }
227
228    Ok(())
229}

231fn show_layer_human(
232    layer_info: &LayerInfo,
233    layer: &crate::storage::layer::Layer,
234    args: &LayersArgs,
235) -> Result<()> {
236    println!("{}", "Layer Analysis".green().bold());
237    println!("{}", "═".repeat(50).green());
238
239    println!("{}: {}", "Layer Hash".bold(), layer_info.hash.cyan());
240    println!("{}: {}", "Type".bold(), layer_info.layer_type);
241    println!("{}: {}", "Generation".bold(), layer_info.generation);
242    println!(
243        "{}: {}",
244        "Parent Hash".bold(),
245        layer_info.parent_hash.cyan()
246    );
247    println!(
248        "{}: {}",
249        "Timestamp".bold(),
250        format_timestamp(layer_info.timestamp)
251    );
252
253    if args.size {
254        println!("\n{}", "Storage Details:".bold());
255        println!("  Layer File Size: {}", format_bytes(layer_info.file_size));
256        println!(
257            "  Uncompressed Size: {}",
258            format_bytes(layer_info.total_file_size)
259        );
260        if layer_info.total_file_size > 0 {
261            let compression_ratio =
262                1.0 - (layer_info.file_size as f64 / layer_info.total_file_size as f64);
263            println!("  Compression Ratio: {:.1}%", compression_ratio * 100.0);
264        }
265        println!("  Scrambling: Enabled (URN-protected)");
266    }
267
268    println!("\n{}", "Content Summary:".bold());
269    println!("  Files: {}", layer_info.files_count);
270    println!(
271        "  Total File Size: {}",
272        format_bytes(layer_info.total_file_size)
273    );
274    println!("  Chunks: {}", layer_info.chunks_count);
275
276    if layer_info.chunks_count > 0 {
277        println!(
278            "  Average Chunk Size: {}",
279            format_bytes(layer_info.total_file_size / layer_info.chunks_count as u64)
280        );
281    }
282
283    if let Some(message) = &layer_info.commit_message {
284        println!("  Commit Message: {}", message.bright_white());
285    }
286
287    if let Some(author) = &layer_info.author {
288        println!("  Author: {}", author);
289    }
290
291    if args.files && !layer.files.is_empty() {
292        println!("\n{}", "Files:".bold());
293        for (i, file) in layer.files.iter().enumerate().take(10) {
294            println!(
295                "  {}. {} ({}, {} chunks)",
296                i + 1,
297                file.path.display().to_string().cyan(),
298                format_bytes(file.size),
299                file.chunks.len()
300            );
301        }
302
303        if layer.files.len() > 10 {
304            println!("  ... and {} more files", layer.files.len() - 10);
305        }
306    }
307
308    if args.chunks && !layer.chunks.is_empty() {
309        println!("\n{}", "Chunk Analysis:".bold());
310
311        // Analyze chunk size distribution
312        let mut size_buckets = [0; 4]; // <1KB, 1KB-32KB, 32KB-1MB, >1MB
313        for chunk in &layer.chunks {
314            let size = chunk.size;
315            if size < 1024 {
316                size_buckets[0] += 1;
317            } else if size < 32768 {
318                size_buckets[1] += 1;
319            } else if size < 1048576 {
320                size_buckets[2] += 1;
321            } else {
322                size_buckets[3] += 1;
323            }
324        }
325
326        println!("  Size Distribution:");
327        println!(
328            "    < 1KB: {} chunks ({:.1}%)",
329            size_buckets[0],
330            size_buckets[0] as f64 / layer.chunks.len() as f64 * 100.0
331        );
332        println!(
333            "    1KB-32KB: {} chunks ({:.1}%)",
334            size_buckets[1],
335            size_buckets[1] as f64 / layer.chunks.len() as f64 * 100.0
336        );
337        println!(
338            "    32KB-1MB: {} chunks ({:.1}%)",
339            size_buckets[2],
340            size_buckets[2] as f64 / layer.chunks.len() as f64 * 100.0
341        );
342        println!(
343            "    > 1MB: {} chunks ({:.1}%)",
344            size_buckets[3],
345            size_buckets[3] as f64 / layer.chunks.len() as f64 * 100.0
346        );
347    }
348
349    Ok(())
350}

352fn show_layer_json(
353    layer_info: &LayerInfo,
354    layer: &crate::storage::layer::Layer,
355    args: &LayersArgs,
356) -> Result<()> {
357    let mut output = json!(layer_info);
358
359    if args.files {
360        output["files"] = json!(layer
361            .files
362            .iter()
363            .map(|f| json!({
364                "path": f.path.display().to_string(),
365                "hash": f.hash.to_hex(),
366                "size": f.size,
367                "chunks": f.chunks.len()
368            }))
369            .collect::<Vec<_>>());
370    }
371
372    if args.chunks {
373        // Analyze chunk size distribution
374        let mut size_buckets = [0; 4];
375        for chunk in &layer.chunks {
376            let size = chunk.size;
377            if size < 1024 {
378                size_buckets[0] += 1;
379            } else if size < 32768 {
380                size_buckets[1] += 1;
381            } else if size < 1048576 {
382                size_buckets[2] += 1;
383            } else {
384                size_buckets[3] += 1;
385            }
386        }
387
388        output["chunk_analysis"] = json!({
389            "total_chunks": layer.chunks.len(),
390            "size_distribution": {
391                "under_1kb": size_buckets[0],
392                "1kb_to_32kb": size_buckets[1],
393                "32kb_to_1mb": size_buckets[2],
394                "over_1mb": size_buckets[3]
395            }
396        });
397    }
398
399    println!("{}", serde_json::to_string_pretty(&output)?);
400    Ok(())
401}

403fn format_timestamp(timestamp: i64) -> String {
404    use std::time::{Duration, UNIX_EPOCH};
405
406    let datetime = UNIX_EPOCH + Duration::from_secs(timestamp as u64);
407    format!("{:?}", datetime)
408        .split_once('.')
409        .map(|(s, _)| s)
410        .unwrap_or("Unknown")
411        .to_string()
412}
413
414fn format_bytes(bytes: u64) -> String {
415    const UNITS: &[&str] = &["B", "KB", "MB", "GB", "TB"];
416    let mut size = bytes as f64;
417    let mut unit_index = 0;
418
419    while size >= 1024.0 && unit_index < UNITS.len() - 1 {
420        size /= 1024.0;
421        unit_index += 1;
422    }
423
424    if unit_index == 0 {
425        format!("{} {}", size as u64, UNITS[unit_index])
426    } else {
427        format!("{:.1} {}", size, UNITS[unit_index])
428    }
429}
