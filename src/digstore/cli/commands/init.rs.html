1//! Initialize command implementation
2
3use anyhow::Result;
4use crate::storage::store::Store;
5use crate::core::types::Hash;
6use std::env;
7use colored::Colorize;
8
9/// Execute the init command
10pub fn execute(
11    store_id: Option<String>,
12    name: Option<String>,
13    no_compression: bool,
14    chunk_size: u32,
15) -> Result<()> {
16    let current_dir = env::current_dir()?;
17    
18    println!("{}", "Initializing repository...".bright_blue());
19    
20    // Check if custom store ID was provided
21    let actual_store_id = if let Some(id_str) = store_id {
22        println!("  {} Using provided store ID: {}", "•".cyan(), id_str.dimmed());
23        Hash::from_hex(&id_str)
24            .map_err(|_| anyhow::anyhow!("Invalid store ID format: {}", id_str))?
25    } else {
26        println!("  {} Generating new store ID...", "•".cyan());
27        crate::storage::store::generate_store_id()
28    };
29    
30    // Initialize the store
31    let store = Store::init(&current_dir)?;
32    
33    println!("  {} Created store directory: {}", 
34        "✓".green(), 
35        store.global_path().display().to_string().dimmed());
36    
37    println!("  {} Generated store ID: {}", 
38        "✓".green(), 
39        store.store_id().to_hex().bright_cyan());
40    
41    println!("  {} Created .digstore file", "✓".green());
42    println!("  {} Initialized empty repository", "✓".green());
43    
44    if let Some(repo_name) = name {
45        println!("  {} Repository name: {}", "•".cyan(), repo_name.bright_white());
46    }
47    
48    if !no_compression {
49        println!("  {} Compression: enabled (zstd)", "•".cyan());
50    } else {
51        println!("  {} Compression: disabled", "•".cyan());
52    }
53    
54    println!("  {} Chunk size: {}KB", "•".cyan(), chunk_size);
55    
56    println!();
57    println!("{}", "Repository initialized".green());
58    println!("Store ID: {}", store.store_id().to_hex().bright_cyan());
59    println!("Location: {}", store.global_path().display().to_string().dimmed());
60    
61    Ok(())
62}
