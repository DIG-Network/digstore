3use crate::cli::context::CliContext;
4use crate::crypto::{derive_key_from_urn, derive_storage_address, transform_urn};
5use crate::wallet::WalletManager;
6use anyhow::Result;
7use colored::Colorize;
8use std::path::PathBuf;

10/// Execute the keygen command
11pub fn execute(
12    urn: String,
13    output: Option<PathBuf>,
14    storage_address: bool,
15    encryption_key: bool,
16    json: bool,
17) -> Result<()> {
18    execute_with_profile(urn, output, storage_address, encryption_key, json, None)
19}
20
21/// Execute the keygen command with optional wallet profile
22pub fn execute_with_profile(
23    urn: String,
24    output: Option<PathBuf>,
25    storage_address: bool,
26    encryption_key: bool,
27    json: bool,
28    wallet_profile: Option<String>,
29) -> Result<()> {
30    println!("{}", "Generating content key...".bright_blue());
31
32    // Get wallet profile from CLI context if not provided
33    let effective_wallet_profile = wallet_profile.or_else(CliContext::get_wallet_profile);
34
35    // Get public key from specified wallet profile or active wallet
36    let public_key =
37        WalletManager::get_wallet_public_key(effective_wallet_profile).map_err(|e| {
38            anyhow::anyhow!(
39                "Failed to get public key from wallet: {}. Ensure you have a wallet initialized.",
40                e
41            )
42        })?;
43
44    let public_key_hex = public_key.to_hex();
45
46    println!("  {} URN: {}", "•".cyan(), urn.dimmed());
47    println!("  {} Public Key: {}", "•".cyan(), public_key_hex.dimmed());
48
49    // Generate transformed address for storage
50    let transformed_address = transform_urn(&urn, &public_key)?;
51    println!(
52        "  {} Transformed Address: {}",
53        "•".cyan(),
54        transformed_address.dimmed()
55    );
56
57    // Generate storage address from transformed URN
58    let storage_addr = derive_storage_address(&urn, &public_key)?;
59
60    // Generate encryption key from original URN
61    let encryption_key_bytes = derive_key_from_urn(&urn);
62    let encryption_key_hex = hex::encode(encryption_key_bytes);
63
64    if json {
65        let output_data = serde_json::json!({
66            "urn": urn,
67            "public_key": public_key_hex,
68            "transformed_address": transformed_address,
69            "storage_address": storage_addr,
70            "encryption_key": encryption_key_hex,
71            "key_derivation": "SHA256(urn)",
72            "address_derivation": "SHA256(transform(urn + public_key))"
73        });
74
75        if let Some(output_path) = &output {
76            std::fs::write(output_path, serde_json::to_string_pretty(&output_data)?)?;
77            println!(
78                "  {} Key information written to: {}",
79                "✓".green(),
80                output_path.display()
81            );
82        } else {
83            println!("{}", serde_json::to_string_pretty(&output_data)?);
84        }
85    } else {
86        println!();
87        println!("{}", "Generated Keys:".green().bold());
88        println!("{}", "═".repeat(50));
89
90        if storage_address || (!storage_address && !encryption_key) {
91            println!("\n{}", "Storage Address:".bold());
92            println!("  Address: {}", storage_addr.bright_cyan());
93            println!("  Purpose: Where encrypted data is stored");
94            println!("  Derivation: SHA256(transform(URN + public_key))");
95        }
96
97        if encryption_key || (!storage_address && !encryption_key) {
98            println!("\n{}", "Encryption Key:".bold());
99            println!("  Key: {}", encryption_key_hex.bright_yellow());
100            println!("  Purpose: Encrypt/decrypt data using AES-256-GCM");
101            println!("  Derivation: SHA256(URN)");
102        }
103
104        println!("\n{}", "URN Transformation:".bold());
105        println!("  Original URN: {}", urn.bright_white());
106        println!("  Transformed:  {}", transformed_address.bright_magenta());
107        println!("  Purpose: Zero-knowledge storage addressing");
108
109        if let Some(output_path) = &output {
110            let output_content = format!(
111                "URN: {}\nPublic Key: {}\nTransformed Address: {}\nStorage Address: {}\nEncryption Key: {}\n",
112                urn, public_key_hex, transformed_address, storage_addr, encryption_key_hex
113            );
114            std::fs::write(output_path, output_content)?;
115            println!(
116                "\n  {} Key information written to: {}",
117                "✓".green(),
118                output_path.display()
119            );
120        }
121    }
122
123    println!();
124    println!("{}", "Security Properties:".yellow().bold());
125    println!("  • Storage layer cannot determine actual URN being used");
126    println!("  • Storage layer cannot decrypt data without original URN");
127    println!("  • Different public keys create different storage addresses");
128    println!("  • Same URN+key always produces same addresses (deterministic)");
129
130    Ok(())
131}
