32pub fn execute(
33    proof: PathBuf,
34    target: Option<String>,
35    root: Option<String>,
36    verbose: bool,
37    from_stdin: bool,
38) -> Result<()> {
39    let args = VerifyArgs {
40        proof,
41        target,
42        root,
43        verbose,
44        from_stdin,
45    };
46
47    println!("{}", "Verifying proof...".green());
48
49    // Read proof data
50    let proof_data = if args.from_stdin {
51        println!("  â€¢ Reading proof from stdin");
52        let mut buffer = String::new();
53        io::stdin().read_to_string(&mut buffer)?;
54        buffer
55    } else {
56        println!("  â€¢ Reading proof from: {}", args.proof.display());
57        std::fs::read_to_string(&args.proof)?
58    };
59
60    // Parse the proof
61    let proof = if proof_data.trim().starts_with('{') {
62        // JSON format
63        Proof::from_json(&proof_data)?
64    } else {
65        return Err(DigstoreError::internal("Only JSON proof format is currently supported").into());
66    };
67
68    if args.verbose {
69        println!("  â€¢ Proof type: {}", proof.proof_type.cyan());
70        println!("  â€¢ Target: {:?}", proof.target);
71        println!("  â€¢ Root hash: {}", proof.root.to_hex().cyan());
72        println!("  â€¢ Proof elements: {}", proof.proof_path.len());
73    }
74
75    // Verify expected values if provided
76    if let Some(expected_target) = &args.target {
77        if args.verbose {
78            println!("  â€¢ Checking target hash...");
79        }
80        // For now, just check if the target contains the expected string
81        let target_str = format!("{:?}", proof.target);
82        if !target_str.contains(expected_target) {
83            println!("{}", "âœ— Target verification failed".red());
84            return Err(DigstoreError::internal("Target hash mismatch").into());
85        }
86        if args.verbose {
87            println!("    âœ“ Target hash matches");
88        }
89    }
90
91    if let Some(expected_root) = &args.root {
92        if args.verbose {
93            println!("  â€¢ Checking root hash...");
94        }
95        let expected_root_hash = crate::core::types::Hash::from_hex(expected_root)
96            .map_err(|_| DigstoreError::internal("Invalid expected root hash format"))?;
97        if proof.root != expected_root_hash {
98            println!("{}", "âœ— Root hash verification failed".red());
99            return Err(DigstoreError::internal("Root hash mismatch").into());
100        }
101        if args.verbose {
102            println!("    âœ“ Root hash matches");
103        }
104    }
105
106    // Verify the proof itself
107    if args.verbose {
108        println!("  â€¢ Verifying merkle proof...");
109    }
110
111    let verification_result = proof.verify()?;
112
113    if verification_result {
114        println!("{}", "âœ“ Proof verification successful!".green());
115        
116        if args.verbose {
117            println!("\n{}", "Verification Details:".bold());
118            println!("  â€¢ Proof format: Valid");
119            println!("  â€¢ Merkle path: Valid");
120            println!("  â€¢ Root reconstruction: Success");
121            println!("  â€¢ Cryptographic integrity: Verified");
122        }
123        
124        println!("\n{}", "ðŸ”’ The proof cryptographically verifies the integrity of the target data.".green());
125    } else {
126        println!("{}", "âœ— Proof verification failed!".red());
127        
128        if args.verbose {
129            println!("\n{}", "Verification Issues:".bold().red());
130            println!("  â€¢ The proof path does not reconstruct to the expected root");
131            println!("  â€¢ This indicates the proof is invalid or corrupted");
132        }
133        
134        return Err(DigstoreError::internal("Proof verification failed").into());
135    }
136
137    Ok(())
138}
