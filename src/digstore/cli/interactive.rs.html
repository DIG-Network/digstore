1//! Interactive CLI prompts for user input
2//!
3//! This module provides interactive prompts for handling various user scenarios
4//! such as store recreation, confirmations, and recovery options.
5
6use anyhow::Result;
7use colored::Colorize;
8use dialoguer::{Confirm, Input};
9use std::path::Path;
10use crate::storage::Store;
11
12/// Ask user if they want to recreate a missing store
13pub fn ask_recreate_store(archive_path: &Path, store_id_hex: &str, auto_yes: bool) -> Result<bool> {
14    if auto_yes {
15        println!("{}", "Store not found - auto-recreating (--yes flag)".green());
16        return Ok(true);
17    }
18    
19    println!();
20    println!("{}", "Store not found!".red().bold());
21    println!("  Archive file: {}", archive_path.display().to_string().yellow());
22    println!("  Store ID: {}", store_id_hex.cyan());
23    println!();
24    
25    let recreate = Confirm::new()
26        .with_prompt("Would you like to recreate this store?")
27        .default(true)
28        .interact()?;
29    
30    Ok(recreate)
31}
32
33/// Ask user for confirmation to overwrite existing .digstore file
34pub fn ask_overwrite_digstore(digstore_path: &Path, auto_yes: bool) -> Result<bool> {
35    if auto_yes {
36        println!("{}", "Repository file exists - auto-overwriting (--yes flag)".green());
37        return Ok(true);
38    }
39    
40    println!();
41    println!("{}", "Warning: Repository file exists".yellow().bold());
42    println!("  File: {}", digstore_path.display().to_string().yellow());
43    println!();
44    println!("{}", "Overwriting will create a new store with a different ID.".red());
45    println!("{}", "Any existing data in the old store will become inaccessible.".red());
46    println!();
47    
48    let overwrite = Confirm::new()
49        .with_prompt("Are you sure you want to overwrite the existing repository?")
50        .default(false)
51        .interact()?;
52    
53    Ok(overwrite)
54}

56/// Interactive store recreation workflow
57pub fn interactive_store_recreation(project_path: &Path) -> Result<Store> {
58    println!();
59    println!("{}", "Creating new repository...".blue().bold());
60    
61    // Get repository name
62    let default_name = project_path
63        .file_name()
64        .and_then(|n| n.to_str())
65        .unwrap_or("my-project");
66    
67    let repo_name: String = Input::new()
68        .with_prompt("Repository name")
69        .default(default_name.to_string())
70        .interact_text()?;
71    
72    println!();
73    println!("Creating repository '{}'...", repo_name.cyan());
74    
75    // Create new store (this will overwrite .digstore if it exists)
76    let store = Store::init(project_path)?;
77    
78    println!();
79    println!("{}", "Repository created".green());
80    println!("  Store ID: {}", store.store_id().to_hex().cyan());
81    println!("  Location: {}", store.global_path().display().to_string().dimmed());
82    
83    Ok(store)
84}

86/// Handle missing store scenario with interactive prompts
87pub fn handle_missing_store(archive_path: &Path, store_id_hex: &str, project_path: &Path, auto_yes: bool) -> Result<Store> {
88    // Ask if user wants to recreate
89    if ask_recreate_store(archive_path, store_id_hex, auto_yes)? {
90        let digstore_path = project_path.join(".digstore");
91        
92        // Check if .digstore file exists and ask for confirmation
93        if digstore_path.exists() {
94            if !ask_overwrite_digstore(&digstore_path, auto_yes)? {
95                println!();
96                println!("{}", "Operation cancelled".yellow());
97                return Err(anyhow::anyhow!("User cancelled store recreation"));
98            }
99        }
100        
101        // Proceed with interactive recreation
102        interactive_store_recreation(project_path)
103    } else {
104        println!();
105        println!("{}", "Operation cancelled".yellow());
106        println!();
107        println!("{}", "Alternative solutions:".blue().bold());
108        println!("  1. {} Check if you're in the right directory", "Location:".green());
109        println!("     {}", "cd /path/to/your/project".cyan());
110        println!();
111        println!("  2. {} Restore the missing store file", "Recovery:".green());
112        println!("     {}", "The store may have been moved or deleted".cyan());
113        println!();
114        
115        Err(anyhow::anyhow!("Store not found and user declined recreation"))
116    }
117}
