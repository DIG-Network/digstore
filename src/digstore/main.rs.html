1//! Digstore Min CLI
2//!
3//! Command-line interface for the Digstore Min content-addressable storage system.
4

20use cli::{
21    context::CliContext, Cli, Commands, LayerCommands, ProofCommands, StagedCommands,
22    StoreCommands, WalletCommands,
23};
24use wallet::WalletManager;

26fn main() -> Result<()> {
27    // Initialize logging
28    tracing_subscriber::fmt()
29        .with_env_filter(
30            tracing_subscriber::EnvFilter::try_from_default_env()
31                .unwrap_or_else(|_| tracing_subscriber::EnvFilter::new("info")),
32        )
33        .init();
34
35    // Parse command line arguments
36    let cli = Cli::parse();
37
38    // Extract custom keys from the command arguments
39    let (custom_encryption_key, custom_decryption_key) = match &cli.command {
40        Commands::Get { decryption_key, .. } => (None, decryption_key.clone()),
41        Commands::Decrypt { decryption_key, .. } => (None, decryption_key.clone()),
42        _ => (None, None),
43    };
44
45    // Set CLI context for commands to access
46    CliContext::set(CliContext {
47        wallet_profile: cli.wallet_profile.clone(),
48        auto_generate_wallet: cli.auto_generate_wallet,
49        auto_import_mnemonic: cli.auto_import_mnemonic.clone(),
50        verbose: cli.verbose,
51        quiet: cli.quiet,
52        yes: cli.yes,
53        custom_encryption_key,
54        custom_decryption_key,
55    });
56
57    // Handle auto-generation or auto-import flags first (these work regardless of command)
58    if cli.auto_generate_wallet || cli.auto_import_mnemonic.is_some() {
59        let wallet_profile = cli.wallet_profile.clone();
60        let mut wallet_manager = WalletManager::new_with_profile(wallet_profile)?;
61
62        if cli.auto_generate_wallet {
63            wallet_manager.auto_generate_wallet()?;
64        } else if let Some(mnemonic) = cli.auto_import_mnemonic.clone() {
65            wallet_manager.auto_import_wallet(&mnemonic)?;
66        }
67    }
68
69    // Check if wallet initialization is needed for this command
70    if needs_wallet_initialization(&cli.command) {
71        let wallet_profile = cli.wallet_profile.clone();
72        let wallet_manager = WalletManager::new_with_profile(wallet_profile)?;
73        wallet_manager.ensure_wallet_initialized()?;
74    }
75
76    // Execute the command
77    match cli.command {
78        Commands::Init {
79            store_id,
80            name,
81            no_compression,
82            chunk_size,
83            encryption_key,
84        } => {
85            cli::commands::init::execute(store_id, name, no_compression, chunk_size, encryption_key)
86        },
87        Commands::Add {
88            paths,
89            recursive,
90            all,
91            force,
92            dry_run,
93            from_stdin,
94            json,
95        } => cli::commands::add::execute(
96            paths, recursive, all, force, dry_run, from_stdin, cli.yes, json,
97        ),
98        Commands::Commit {
99            message,
100            full_layer,
101            author,
102            date,
103            edit,
104            json,
105        } => cli::commands::commit::execute(message, full_layer, author, date, edit, json),
106        Commands::Status {
107            short,
108            porcelain,
109            show_chunks,
110            json,
111        } => cli::commands::status::execute(short, porcelain, show_chunks, json),
112        Commands::Get {
113            path,
114            output,
115            verify,
116            metadata,
117            at,
118            progress,
119            decryption_key,
120            json,
121        } => {
122            // Update CLI context with decryption key before opening stores
123            if let Some(key) = &decryption_key {
124                let mut current_context = CliContext::get().unwrap_or_default();
125                current_context.custom_decryption_key = Some(key.clone());
126                CliContext::set(current_context);
127            }
128            cli::commands::get::execute(
129                path,
130                output,
131                verify,
132                metadata,
133                at,
134                progress,
135                decryption_key,
136                json,
137            )
138        },
139        Commands::Decrypt {
140            path,
141            output,
142            urn,
143            decryption_key,
144            json,
145        } => cli::commands::decrypt::execute(path, output, urn, decryption_key, json),
146        Commands::Keygen {
147            urn,
148            output,
149            storage_address,
150            encryption_key,
151            json,
152        } => cli::commands::keygen::execute(urn, output, storage_address, encryption_key, json),
153        Commands::Cat {
154            path,
155            at,
156            number,
157            no_pager,
158            bytes,
159            json,
160        } => cli::commands::cat::execute(path, at, number, no_pager, bytes, json),
161        Commands::Completion { shell } => cli::commands::completion::execute(shell),
162
163        Commands::Staged { command } => match command {
164            StagedCommands::List {
165                limit,
166                page,
167                detailed,
168                json,
169                all,
170            } => cli::commands::staged::execute_list(limit, page, detailed, json, all),
171            StagedCommands::Diff {
172                name_only,
173                json,
174                stat,
175                unified,
176                file,
177            } => cli::commands::staged::execute_diff(name_only, json, stat, unified, file),
178            StagedCommands::Clear { json, force } => {
179                cli::commands::staged::clear_staged(json, force)
180            },
181        },
182
183        Commands::Layer { command } => match command {
184            LayerCommands::List {
185                json,
186                size,
187                files,
188                chunks,
189            } => cli::commands::layer::execute_list(None, json, true, size, files, chunks),
190            LayerCommands::Analyze {
191                layer_hash,
192                json,
193                size,
194                files,
195                chunks,
196            } => cli::commands::layer::execute_list(
197                Some(layer_hash),
198                json,
199                false,
200                size,
201                files,
202                chunks,
203            ),
204            LayerCommands::Inspect {
205                layer_hash,
206                json,
207                header,
208                merkle,
209                chunks,
210                verify,
211            } => cli::commands::layer::execute_inspect(
212                layer_hash, json, header, merkle, chunks, verify,
213            ),
214        },
215
216        Commands::Store { command } => match command {
217            StoreCommands::Info {
218                json,
219                config,
220                paths,
221                layer,
222            } => {
223                if layer.is_some() {
224                    cli::commands::store::execute_info(json, layer)
225                } else {
226                    cli::commands::store::execute_store_info(json, config, paths)
227                }
228            },
229            StoreCommands::Log {
230                limit,
231                oneline,
232                graph,
233                since,
234            } => cli::commands::store::execute_log(limit, oneline, graph, since),
235            StoreCommands::History {
236                json,
237                limit,
238                stats,
239                graph,
240                since,
241            } => cli::commands::store::execute_history(json, limit, stats, graph, since),
242            StoreCommands::Root {
243                json,
244                verbose,
245                hash_only,
246            } => cli::commands::store::execute_root(json, verbose, hash_only),
247            StoreCommands::Size {
248                json,
249                breakdown,
250                efficiency,
251                layers,
252            } => cli::commands::store::execute_size(json, breakdown, efficiency, layers),
253            StoreCommands::Stats {
254                json,
255                detailed,
256                performance,
257                security,
258            } => cli::commands::store::execute_stats(json, detailed, performance, security),
259        },
260
261        Commands::Proof { command } => match command {
262            ProofCommands::Generate {
263                target,
264                output,
265                format,
266                at,
267                bytes,
268                compact,
269            } => cli::commands::proof::execute_generate(target, output, format, at, bytes, compact),
270            ProofCommands::Verify {
271                proof,
272                target,
273                root,
274                verbose,
275                from_stdin,
276            } => cli::commands::proof::execute_verify(proof, target, root, verbose, from_stdin),
277            ProofCommands::GenerateArchiveSize {
278                store_id,
279                output,
280                format,
281                verbose,
282                show_compression,
283                json,
284            } => cli::commands::proof::execute_generate_archive_size(
285                store_id,
286                output,
287                format,
288                verbose,
289                show_compression,
290                json,
291            ),
292            ProofCommands::VerifyArchiveSize {
293                proof,
294                store_id,
295                root_hash,
296                expected_size,
297                publisher_public_key,
298                from_file,
299                verbose,
300                json,
301            } => cli::commands::proof::execute_verify_archive_size(
302                proof,
303                store_id,
304                root_hash,
305                expected_size,
306                publisher_public_key,
307                from_file,
308                verbose,
309                json,
310            ),
311        },
312
313        Commands::Config {
314            key,
315            value,
316            list,
317            unset,
318            show_origin,
319            edit,
320            json,
321        } => cli::commands::config::execute(key, value, list, unset, show_origin, edit, json),
322
323        Commands::Wallet { command } => {
324            cli::commands::wallet::execute(command).map_err(|e| e.into())
325        },
326    }
327}
328
329/// Determines if a command requires wallet initialization
330fn needs_wallet_initialization(command: &Commands) -> bool {
331    match command {
332        // Commands that don't need wallet initialization
333        Commands::Completion { .. } => false,
334        Commands::Config { .. } => false,
335        Commands::Wallet { .. } => false, // Wallet commands manage their own initialization
336
337        // All other commands require wallet initialization
338        _ => true,
339    }
340}
