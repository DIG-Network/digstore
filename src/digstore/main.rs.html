1//! Digstore Min CLI
2//!
3//! Command-line interface for the Digstore Min content-addressable storage system.
4
5use anyhow::Result;
6use clap::Parser;
7use tracing_subscriber;
8
9mod cli;
10mod config;
11mod core;
12mod ignore;
13mod proofs;
14mod security;
15mod storage;
16mod urn;
17
18use cli::{Cli, Commands, LayerCommands, ProofCommands, StagedCommands, StoreCommands};
19
20fn main() -> Result<()> {
21    // Initialize logging
22    tracing_subscriber::fmt()
23        .with_env_filter(
24            tracing_subscriber::EnvFilter::try_from_default_env()
25                .unwrap_or_else(|_| tracing_subscriber::EnvFilter::new("info")),
26        )
27        .init();
28
29    // Parse command line arguments
30    let cli = Cli::parse();
31
32    // Execute the command
33    match cli.command {
34        Commands::Init {
35            store_id,
36            name,
37            no_compression,
38            chunk_size,
39        } => cli::commands::init::execute(store_id, name, no_compression, chunk_size),
40        Commands::Add {
41            paths,
42            recursive,
43            all,
44            force,
45            dry_run,
46            from_stdin,
47            json,
48        } => cli::commands::add::execute(
49            paths, recursive, all, force, dry_run, from_stdin, cli.yes, json,
50        ),
51        Commands::Commit {
52            message,
53            full_layer,
54            author,
55            date,
56            edit,
57            json,
58        } => cli::commands::commit::execute(message, full_layer, author, date, edit, json),
59        Commands::Status {
60            short,
61            porcelain,
62            show_chunks,
63            json,
64        } => cli::commands::status::execute(short, porcelain, show_chunks, json),
65        Commands::Get {
66            path,
67            output,
68            verify,
69            metadata,
70            at,
71            progress,
72            json,
73        } => cli::commands::get::execute(path, output, verify, metadata, at, progress, json),
74        Commands::Cat {
75            path,
76            at,
77            number,
78            no_pager,
79            bytes,
80            json,
81        } => cli::commands::cat::execute(path, at, number, no_pager, bytes, json),
82        Commands::Completion { shell } => cli::commands::completion::execute(shell),
83
84        Commands::Staged { command } => match command {
85            StagedCommands::List {
86                limit,
87                page,
88                detailed,
89                json,
90                all,
91            } => cli::commands::staged::execute_list(limit, page, detailed, json, all),
92            StagedCommands::Diff {
93                name_only,
94                json,
95                stat,
96                unified,
97                file,
98            } => cli::commands::staged::execute_diff(name_only, json, stat, unified, file),
99            StagedCommands::Clear { json, force } => {
100                cli::commands::staged::clear_staged(json, force)
101            }
102        },
103
104        Commands::Layer { command } => match command {
105            LayerCommands::List {
106                json,
107                size,
108                files,
109                chunks,
110            } => cli::commands::layer::execute_list(None, json, true, size, files, chunks),
111            LayerCommands::Analyze {
112                layer_hash,
113                json,
114                size,
115                files,
116                chunks,
117            } => cli::commands::layer::execute_list(
118                Some(layer_hash),
119                json,
120                false,
121                size,
122                files,
123                chunks,
124            ),
125            LayerCommands::Inspect {
126                layer_hash,
127                json,
128                header,
129                merkle,
130                chunks,
131                verify,
132            } => cli::commands::layer::execute_inspect(
133                layer_hash, json, header, merkle, chunks, verify,
134            ),
135        },
136
137        Commands::Store { command } => match command {
138            StoreCommands::Info {
139                json,
140                config,
141                paths,
142                layer,
143            } => {
144                if layer.is_some() {
145                    cli::commands::store::execute_info(json, layer)
146                } else {
147                    cli::commands::store::execute_store_info(json, config, paths)
148                }
149            }
150            StoreCommands::Log {
151                limit,
152                oneline,
153                graph,
154                since,
155            } => cli::commands::store::execute_log(limit, oneline, graph, since),
156            StoreCommands::History {
157                json,
158                limit,
159                stats,
160                graph,
161                since,
162            } => cli::commands::store::execute_history(json, limit, stats, graph, since),
163            StoreCommands::Root {
164                json,
165                verbose,
166                hash_only,
167            } => cli::commands::store::execute_root(json, verbose, hash_only),
168            StoreCommands::Size {
169                json,
170                breakdown,
171                efficiency,
172                layers,
173            } => cli::commands::store::execute_size(json, breakdown, efficiency, layers),
174            StoreCommands::Stats {
175                json,
176                detailed,
177                performance,
178                security,
179            } => cli::commands::store::execute_stats(json, detailed, performance, security),
180        },
181
182        Commands::Proof { command } => match command {
183            ProofCommands::Generate {
184                target,
185                output,
186                format,
187                at,
188                bytes,
189                compact,
190            } => cli::commands::proof::execute_generate(target, output, format, at, bytes, compact),
191            ProofCommands::Verify {
192                proof,
193                target,
194                root,
195                verbose,
196                from_stdin,
197            } => cli::commands::proof::execute_verify(proof, target, root, verbose, from_stdin),
198        },
199
200        Commands::Config {
201            key,
202            value,
203            list,
204            unset,
205            show_origin,
206            edit,
207            json,
208        } => cli::commands::config::execute(key, value, list, unset, show_origin, edit, json),
209    }
210}
