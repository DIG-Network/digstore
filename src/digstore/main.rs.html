1//! Digstore Min CLI
2//!
3//! Command-line interface for the Digstore Min content-addressable storage system.
4
5use clap::Parser;
6use anyhow::Result;
7use tracing_subscriber;
8
9mod core;
10mod storage;
11mod proofs;
12mod urn;
13mod cli;
14mod security;
15mod ignore;
16mod config;
17
18use cli::{Cli, Commands, StagedCommands, LayerCommands, StoreCommands, ProofCommands};
19
20fn main() -> Result<()> {
21    // Initialize logging
22    tracing_subscriber::fmt()
23        .with_env_filter(
24            tracing_subscriber::EnvFilter::try_from_default_env()
25                .unwrap_or_else(|_| tracing_subscriber::EnvFilter::new("info"))
26        )
27        .init();
28
29    // Parse command line arguments
30    let cli = Cli::parse();
31    
32    // Execute the command
33    match cli.command {
34        Commands::Init { store_id, name, no_compression, chunk_size } => {
35            cli::commands::init::execute(store_id, name, no_compression, chunk_size)
36        }
37        Commands::Add { paths, recursive, all, force, dry_run, from_stdin, json } => {
38            cli::commands::add::execute(paths, recursive, all, force, dry_run, from_stdin, cli.yes, json)
39        }
40        Commands::Commit { message, full_layer, author, date, edit, json } => {
41            cli::commands::commit::execute(message, full_layer, author, date, edit, json)
42        }
43        Commands::Status { short, porcelain, show_chunks, json } => {
44            cli::commands::status::execute(short, porcelain, show_chunks, json)
45        }
46        Commands::Get { path, output, verify, metadata, at, progress, json } => {
47            cli::commands::get::execute(path, output, verify, metadata, at, progress, json)
48        }
49        Commands::Cat { path, at, number, no_pager, bytes, json } => {
50            cli::commands::cat::execute(path, at, number, no_pager, bytes, json)
51        }
52        Commands::Completion { shell } => {
53            cli::commands::completion::execute(shell)
54        }
55
56        Commands::Staged { command } => {
57            match command {
58                StagedCommands::List { limit, page, detailed, json, all } => {
59                    cli::commands::staged::execute_list(limit, page, detailed, json, all)
60                }
61                StagedCommands::Diff { name_only, json, stat, unified, file } => {
62                    cli::commands::staged::execute_diff(name_only, json, stat, unified, file)
63                }
64                StagedCommands::Clear { json, force } => {
65                    cli::commands::staged::clear_staged(json, force)
66                }
67            }
68        }
69
70        Commands::Layer { command } => {
71            match command {
72                LayerCommands::List { json, size, files, chunks } => {
73                    cli::commands::layer::execute_list(None, json, true, size, files, chunks)
74                }
75                LayerCommands::Analyze { layer_hash, json, size, files, chunks } => {
76                    cli::commands::layer::execute_list(Some(layer_hash), json, false, size, files, chunks)
77                }
78                LayerCommands::Inspect { layer_hash, json, header, merkle, chunks, verify } => {
79                    cli::commands::layer::execute_inspect(layer_hash, json, header, merkle, chunks, verify)
80                }
81            }
82        }
83
84        Commands::Store { command } => {
85            match command {
86                StoreCommands::Info { json, config, paths, layer } => {
87                    if layer.is_some() {
88                        cli::commands::store::execute_info(json, layer)
89                    } else {
90                        cli::commands::store::execute_store_info(json, config, paths)
91                    }
92                }
93                StoreCommands::Log { limit, oneline, graph, since } => {
94                    cli::commands::store::execute_log(limit, oneline, graph, since)
95                }
96                StoreCommands::History { json, limit, stats, graph, since } => {
97                    cli::commands::store::execute_history(json, limit, stats, graph, since)
98                }
99                StoreCommands::Root { json, verbose, hash_only } => {
100                    cli::commands::store::execute_root(json, verbose, hash_only)
101                }
102                StoreCommands::Size { json, breakdown, efficiency, layers } => {
103                    cli::commands::store::execute_size(json, breakdown, efficiency, layers)
104                }
105                StoreCommands::Stats { json, detailed, performance, security } => {
106                    cli::commands::store::execute_stats(json, detailed, performance, security)
107                }
108            }
109        }
110
111        Commands::Proof { command } => {
112            match command {
113                ProofCommands::Generate { target, output, format, at, bytes, compact } => {
114                    cli::commands::proof::execute_generate(target, output, format, at, bytes, compact)
115                }
116                ProofCommands::Verify { proof, target, root, verbose, from_stdin } => {
117                    cli::commands::proof::execute_verify(proof, target, root, verbose, from_stdin)
118                }
119            }
120        }
121
122        Commands::Config { key, value, list, unset, show_origin, edit, json } => {
123            cli::commands::config::execute(key, value, list, unset, show_origin, edit, json)
124        }
125    }
126}
