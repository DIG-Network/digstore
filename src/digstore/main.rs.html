38fn main() -> Result<()> {
39    // Initialize logging
40    tracing_subscriber::fmt()
41        .with_env_filter(
42            tracing_subscriber::EnvFilter::try_from_default_env()
43                .unwrap_or_else(|_| tracing_subscriber::EnvFilter::new("info")),
44        )
45        .init();
46
47    // Parse command line arguments
48    let cli = Cli::parse();
49
50    // Extract custom keys from the command arguments
51    let (custom_encryption_key, custom_decryption_key) = match &cli.command {
52        Commands::Decrypt { decryption_key, .. } => (None, decryption_key.clone()),
53        _ => (None, None),
54    };
55
56    // Set CLI context for commands to access
57    CliContext::set(CliContext {
58        wallet_profile: cli.wallet_profile.clone(),
59        auto_generate_wallet: cli.auto_generate_wallet,
60        auto_import_mnemonic: cli.auto_import_mnemonic.clone(),
61        verbose: cli.verbose,
62        quiet: cli.quiet,
63        yes: cli.yes,
64        non_interactive: cli.non_interactive,
65        custom_encryption_key,
66        custom_decryption_key,
67    });
68
69    // Handle auto-generation or auto-import flags first (these work regardless of command)
70    if cli.auto_generate_wallet || cli.auto_import_mnemonic.is_some() {
71        let wallet_profile = cli.wallet_profile.clone();
72        let wallet_manager = WalletManager::new_with_profile(wallet_profile)?;
73
74        if cli.auto_generate_wallet {
75            wallet_manager.auto_generate_wallet()?;
76        } else if let Some(mnemonic) = cli.auto_import_mnemonic.clone() {
77            wallet_manager.auto_import_wallet(&mnemonic)?;
78        }
79    }
80
81    // Check if wallet initialization is needed for this command
82    if needs_wallet_initialization(&cli.command) {
83        let wallet_profile = cli.wallet_profile.clone();
84        let wallet_manager = WalletManager::new_with_profile(wallet_profile)?;
85        wallet_manager.ensure_wallet_initialized()?;
86    }
87
88    // Check for updates (unless running update command, in quiet mode, or non-interactive)
89    if !matches!(cli.command, Commands::Update { .. }) && !cli.quiet && !cli.non_interactive {
90        check_and_prompt_for_updates()?;
91    }
92
93    // Execute the command
94    match cli.command {
95        Commands::Init {
96            name,
97            encryption_key,
98        } => {
99            cli::commands::init::execute(name, encryption_key)
100        },
101        Commands::Add {
102            paths,
103            recursive,
104            all,
105            force,
106            dry_run,
107            from_stdin,
108            json,
109        } => cli::commands::add::execute(
110            paths, recursive, all, force, dry_run, from_stdin, cli.yes, json,
111        ),
112        Commands::Commit {
113            message,
114            author,
115            date,
116            edit,
117            json,
118        } => cli::commands::commit::execute(message, author, date, edit, json),
119        Commands::Status {
120            short,
121            porcelain,
122            show_chunks,
123            json,
124        } => cli::commands::status::execute(short, porcelain, show_chunks, json),
125        Commands::Get {
126            path,
127            output,
128            verify,
129            metadata,
130            at,
131            progress,
132            json,
133        } => {
134            cli::commands::get::execute(
135                path,
136                output,
137                verify,
138                metadata,
139                at,
140                progress,
141                json,
142            )
143        },
144        Commands::Decrypt {
145            path,
146            output,
147            urn,
148            decryption_key,
149            json,
150        } => cli::commands::decrypt::execute(path, output, urn, decryption_key, json),
151        Commands::Keygen {
152            urn,
153            output,
154            storage_address,
155            encryption_key,
156            json,
157        } => cli::commands::keygen::execute(urn, output, storage_address, encryption_key, json),
158        Commands::Completion { shell } => cli::commands::completion::execute(shell),
159
160        Commands::Staged { command } => match command {
161            StagedCommands::List {
162                limit,
163                page,
164                detailed,
165                json,
166                all,
167            } => cli::commands::staged::execute_list(limit, page, detailed, json, all),
168            StagedCommands::Diff {
169                name_only,
170                json,
171                stat,
172                unified,
173                file,
174            } => cli::commands::staged::execute_diff(name_only, json, stat, unified, file),
175            StagedCommands::Clear { json, force } => {
176                cli::commands::staged::clear_staged(json, force)
177            },
178        },
179
180        Commands::Update {
181            check_only,
182            force,
183            json,
184        } => cli::commands::update::execute(check_only, force, json),
185
186        Commands::Version { subcommand, version } => {
187            cli::commands::version_cmd::execute(subcommand, version)
188                .map_err(|e| anyhow::anyhow!("Version command failed: {}", e))
189        },
190
191        Commands::Layer { command } => match command {
192            LayerCommands::List {
193                json,
194                size,
195                files,
196                chunks,
197            } => cli::commands::layer::execute_list(None, json, true, size, files, chunks),
198            LayerCommands::Analyze {
199                layer_hash,
200                json,
201                size,
202                files,
203                chunks,
204            } => cli::commands::layer::execute_list(
205                Some(layer_hash),
206                json,
207                false,
208                size,
209                files,
210                chunks,
211            ),
212            LayerCommands::Inspect {
213                layer_hash,
214                json,
215                header,
216                merkle,
217                chunks,
218                verify,
219            } => cli::commands::layer::execute_inspect(
220                layer_hash, json, header, merkle, chunks, verify,
221            ),
222        },
223
224        Commands::Store { command } => match command {
225            StoreCommands::Info {
226                json,
227                config,
228                paths,
229                layer,
230            } => {
231                if layer.is_some() {
232                    cli::commands::store::execute_info(json, layer)
233                } else {
234                    cli::commands::store::execute_store_info(json, config, paths)
235                }
236            },
237            StoreCommands::Log {
238                limit,
239                oneline,
240                graph,
241                since,
242            } => cli::commands::store::execute_log(limit, oneline, graph, since),
243            StoreCommands::History {
244                json,
245                limit,
246                stats,
247                graph,
248                since,
249            } => cli::commands::store::execute_history(json, limit, stats, graph, since),
250            StoreCommands::Root {
251                json,
252                verbose,
253                hash_only,
254            } => cli::commands::store::execute_root(json, verbose, hash_only),
255            StoreCommands::Size {
256                json,
257                breakdown,
258                efficiency,
259                layers,
260            } => cli::commands::store::execute_size(json, breakdown, efficiency, layers),
261            StoreCommands::Stats {
262                json,
263                detailed,
264                performance,
265                security,
266            } => cli::commands::store::execute_stats(json, detailed, performance, security),
267        },
268
269        Commands::Proof { command } => match command {
270            ProofCommands::Generate {
271                target,
272                output,
273                format,
274                at,
275                bytes,
276            } => cli::commands::proof::execute_generate(target, output, format, at, bytes),
277            ProofCommands::Verify {
278                proof,
279                target,
280                root,
281                verbose,
282                from_stdin,
283            } => cli::commands::proof::execute_verify(proof, target, root, verbose, from_stdin),
284            ProofCommands::GenerateArchiveSize {
285                store_id,
286                output,
287                format,
288                verbose,
289                show_compression,
290                json,
291            } => cli::commands::proof::execute_generate_archive_size(
292                store_id,
293                output,
294                format,
295                verbose,
296                show_compression,
297                json,
298            ),
299            ProofCommands::VerifyArchiveSize {
300                proof,
301                store_id,
302                root_hash,
303                expected_size,
304                publisher_public_key,
305                from_file,
306                verbose,
307                json,
308            } => cli::commands::proof::execute_verify_archive_size(
309                proof,
310                store_id,
311                root_hash,
312                expected_size,
313                publisher_public_key,
314                from_file,
315                verbose,
316                json,
317            ),
318        },
319
320        Commands::Config {
321            key,
322            value,
323            list,
324            unset,
325            show_origin,
326            edit,
327            json,
328        } => cli::commands::config::execute(key, value, list, unset, show_origin, edit, json),
329
330        Commands::Wallet { command } => {
331            cli::commands::wallet::execute(command).map_err(|e| e.into())
332        },
333    }
334}

336/// Determines if a command requires wallet initialization
337fn needs_wallet_initialization(command: &Commands) -> bool {
338    match command {
339        // Commands that don't need wallet initialization
340        Commands::Completion { .. } => false,
341        Commands::Config { .. } => false,
342        Commands::Update { .. } => false,
343        Commands::Version { .. } => false,
344        Commands::Wallet { .. } => false, // Wallet commands manage their own initialization
345
346        // All other commands require wallet initialization
347        _ => true,
348    }
349}

351/// Check for updates and prompt user if available
352fn check_and_prompt_for_updates() -> Result<()> {
353    use colored::Colorize;
354    use dialoguer::Confirm;
355
356    // Only check occasionally to avoid slowing down commands
357    let update_info = match crate::update::check_for_updates() {
358        Ok(info) => info,
359        Err(_) => return Ok(()), // Silently ignore update check failures
360    };
361
362    if !update_info.update_available {
363        return Ok(());
364    }
365
366    println!();
367    println!(
368        "{} {} Update available: {} â†’ {}",
369        "ðŸŽ‰".bright_yellow(),
370        "Digstore".bright_cyan().bold(),
371        update_info.current_version.dimmed(),
372        update_info.latest_version.bright_green()
373    );
374
375    let should_update = if crate::cli::context::CliContext::is_non_interactive() {
376        false // Don't auto-update in non-interactive mode
377    } else {
378        Confirm::new()
379            .with_prompt("Would you like to download and install the update now?")
380            .default(false)
381            .interact()
382            .unwrap_or(false)
383    };
384
385    if should_update {
386        println!();
387        if let Some(download_url) = update_info.download_url {
388            match crate::update::download_and_install_update(&download_url) {
389                Ok(_) => {
390                    println!(
391                        "{}",
392                        "âœ“ Update completed! Please restart your terminal."
393                            .green()
394                            .bold()
395                    );
396                },
397                Err(e) => {
398                    println!("{} Update failed: {}", "âœ—".red(), e);
399                    println!(
400                        "  {} Manual download: https://github.com/DIG-Network/digstore/releases",
401                        "â†’".cyan()
402                    );
403                },
404            }
405        }
406    } else {
407        println!("  {} Use 'digstore update' to update later", "â†’".cyan());
408    }
409
410    println!();
411    Ok(())
