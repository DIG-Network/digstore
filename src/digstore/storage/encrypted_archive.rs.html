16/// Wrapper around DigArchive that handles address transformation
17pub struct EncryptedArchive {

20    custom_encryption_key: Option<String>,

30    /// Create new encrypted archive wrapper with store ID for config loading
31    pub fn new_with_store_id(
32        archive: DigArchive,
33        store_id: Option<&crate::core::types::StoreId>,
34    ) -> Result<Self> {
35        let config = GlobalConfig::load()?;
36
37        // Try to get public key from wallet (using CLI context profile), fallback to config
38        let wallet_profile = CliContext::get_wallet_profile();
39        let public_key = match WalletManager::get_wallet_public_key(wallet_profile) {
40            Ok(wallet_key) => Some(wallet_key),
41            Err(_) => {
42                // Fallback to config for backward compatibility
43                if let Some(hex_key) = config.crypto.public_key {
44                    Some(PublicKey::from_hex(&hex_key)?)
45                } else {
46                    None
47                }
48            },
49        };
50
51        let encrypted_storage = config.crypto.encrypted_storage.unwrap_or(true);
52
53        // Check for custom encryption/decryption key from multiple sources:
54        // 1. CLI context (for current command)
55        // 2. Store config (for persistent store-wide encryption)
56        let custom_encryption_key = CliContext::get_custom_encryption_key()
57            .or_else(|| CliContext::get_custom_decryption_key())
58            .or_else(|| {
59                // Load from store config if available
60                if let Some(store_id) = store_id {
61                    crate::config::StoreConfig::load(store_id)
62                        .ok()
63                        .and_then(|config| config.get_custom_encryption_key())
64                } else {
65                    None
66                }
67            });
68
69        Ok(Self {
70            archive,
71            public_key,
72            custom_encryption_key,
73            encrypted_storage,
74        })
75    }

103    /// Transform a layer hash for storage
104    fn transform_layer_hash(&self, layer_hash: &Hash) -> Result<Hash> {
105        if let Some(custom_key) = &self.custom_encryption_key {
106            // For custom encryption, use the key directly as a transformation seed
107            use sha2::{Digest, Sha256};
108            let mut hasher = Sha256::new();
109            hasher.update(layer_hash.as_bytes());
110            hasher.update(custom_key.as_bytes());
111            hasher.update(b"layer_transform");
112            let transformed_bytes = hasher.finalize();
113            Ok(Hash::from_bytes(transformed_bytes.into()))
114        } else if let Some(public_key) = &self.public_key {
115            // Create URN from layer hash
116            let layer_urn = format!("urn:dig:layer:{}", layer_hash.to_hex());
117
118            // Transform the URN (returns raw hex string, not URN format)
119            let transformed_hex = transform_urn(&layer_urn, public_key)?;
120
121            // Parse the hex string directly as the new hash
122            Hash::from_hex(&transformed_hex)
123                .map_err(|_| DigstoreError::internal("Invalid hash from URN transformation"))
124        } else {
125            Ok(*layer_hash)
126        }
127    }

129    /// Add a layer with optional transformation
130    pub fn add_layer(&mut self, layer_hash: Hash, layer_data: &[u8]) -> Result<()> {
131        // Set custom encryption flag in archive header if using custom key
132        if self.custom_encryption_key.is_some() {
133            self.archive.set_custom_encryption(true)?;
134        }
135
136        let storage_hash = if self.is_encrypted() {
137            self.transform_layer_hash(&layer_hash)?
138        } else {
139            layer_hash
140        };
141
142        self.archive.add_layer(storage_hash, layer_data)
143    }

145    /// Get a layer with optional transformation
146    pub fn get_layer(&self, layer_hash: &Hash) -> Result<Layer> {
147        let storage_hash = if self.is_encrypted() {
148            self.transform_layer_hash(layer_hash)?
149        } else {
150            *layer_hash
151        };
152
153        // Try to get the layer, but if it fails and encryption is enabled,
154        // return deterministic random data instead of error
155        match self.archive.get_layer(&storage_hash) {
156            Ok(layer) => Ok(layer),
157            Err(_) if self.is_encrypted() => {
158                // Generate deterministic random layer data based on the original layer hash
159                self.generate_random_layer_data(layer_hash)
160            },
161            Err(e) => Err(e),
162        }
163    }

221    /// Generate a deterministic random file size that looks realistic
222    /// This produces sizes that follow common file size patterns to make decoys indistinguishable
223    /// from real content based on size alone.
224    fn generate_deterministic_random_size(&self, seed: &str) -> usize {
225        use sha2::{Digest, Sha256};
226
227        let mut hasher = Sha256::new();
228        hasher.update(seed.as_bytes());
229        hasher.update(b"size_generation");
230        let hash = hasher.finalize();
231
232        // Use first 8 bytes as a u64 for randomness
233        let mut bytes = [0u8; 8];
234        bytes.copy_from_slice(&hash[0..8]);
235        let random_value = u64::from_le_bytes(bytes);
236
237        // Create realistic file size distribution:
238        // 40% small files (1KB - 100KB)
239        // 35% medium files (100KB - 1MB)
240        // 20% large files (1MB - 10MB)
241        // 5% very large files (10MB - 20MB)
242
243        let size_category = random_value % 100;
244        let size_random = (random_value >> 8) % 1000000; // Use remaining bits for size within category
245
246        match size_category {
247            0..=39 => {
248                // Small files: 1KB - 100KB
249                let base = 1024; // 1KB
250                let range = 99 * 1024; // up to 100KB
251                base + (size_random % range) as usize
252            },
253            40..=74 => {
254                // Medium files: 100KB - 1MB
255                let base = 100 * 1024; // 100KB
256                let range = 924 * 1024; // up to 1MB
257                base + (size_random % range as u64) as usize
258            },
259            75..=94 => {
260                // Large files: 1MB - 10MB
261                let base = 1024 * 1024; // 1MB
262                let range = 9 * 1024 * 1024; // up to 10MB
263                base + (size_random % range as u64) as usize
264            },
265            _ => {
266                // Very large files: 10MB - 20MB
267                let base = 10 * 1024 * 1024; // 10MB
268                let range = 10 * 1024 * 1024; // up to 20MB
269                base + (size_random % range as u64) as usize
270            },
271        }
272    }

274    /// Generate deterministic random data for invalid content addresses
275    fn generate_random_data_for_hash(&self, layer_hash: &Hash) -> Vec<u8> {
276        use sha2::{Digest, Sha256};
277
278        // Use the layer hash as seed for deterministic random generation
279        let seed = format!("invalid_content_address:{}", layer_hash.to_hex());
280        let random_size = self.generate_deterministic_random_size(&seed);
281
282        let mut result = Vec::with_capacity(random_size);
283        let mut hasher = Sha256::new();
284        hasher.update(seed.as_bytes());
285        let mut counter = 0u64;
286
287        while result.len() < random_size {
288            let mut current_hasher = hasher.clone();
289            current_hasher.update(&counter.to_le_bytes());
290            let hash = current_hasher.finalize();
291
292            let bytes_needed = random_size - result.len();
293            let bytes_to_copy = bytes_needed.min(hash.len());
294            result.extend_from_slice(&hash[..bytes_to_copy]);
295
296            counter += 1;
297        }
298
299        result
300    }

302    /// Generate deterministic random layer for invalid content addresses
303    fn generate_random_layer_data(&self, layer_hash: &Hash) -> Result<Layer> {
304        use crate::core::types::{FileEntry, FileMetadata, LayerMetadata, LayerType};
305        use std::path::PathBuf;
306
307        // Create a fake layer with random data that appears legitimate
308        let mut layer = Layer::new(crate::core::types::LayerType::Full, 1, Hash::zero());
309
310        // Add fake file entries with deterministic random properties
311        let random_data = self.generate_random_data_for_hash(layer_hash);
312        let chunk_size = 64 * 1024; // 64KB chunks
313
314        for i in 0..(random_data.len() / chunk_size).min(10) {
315            let start = i * chunk_size;
316            let end = (start + chunk_size).min(random_data.len());
317            let chunk_data = &random_data[start..end];
318
319            let chunk_hash = crate::core::hash::sha256(chunk_data);
320            let fake_file_path = PathBuf::from(format!("fake_file_{}.dat", i));
321
322            let file_entry = FileEntry {
323                path: fake_file_path,
324                hash: chunk_hash,
325                size: chunk_data.len() as u64,
326                chunks: vec![crate::core::types::ChunkRef {
327                    hash: chunk_hash,
328                    offset: 0,
329                    size: chunk_data.len() as u32,
330                }],
331                metadata: FileMetadata {
332                    mode: 0o644,
333                    modified: chrono::Utc::now().timestamp(),
334                    is_new: true,
335                    is_modified: false,
336                    is_deleted: false,
337                },
338            };
339
340            layer.add_file(file_entry);
341
342            let chunk = crate::core::types::Chunk {
343                hash: chunk_hash,
344                offset: 0,
345                size: chunk_data.len() as u32,
346                data: chunk_data.to_vec(),
347            };
348
349            layer.add_chunk(chunk);
350        }
351
352        // Set fake metadata
353        layer.metadata.message = Some("Generated random layer".to_string());
354        layer.metadata.author = Some("system".to_string());
355
356        Ok(layer)
357    }
