55    /// Initialize a new store with options
56    pub fn init_with_options(project_path: &Path, auto_yes: bool) -> Result<Self> {
57        // Check if already initialized
58        let digstore_path = project_path.join(".digstore");
59        if digstore_path.exists() {
60            // Ask for confirmation to overwrite
61            use crate::cli::interactive;
62
63            if !interactive::ask_overwrite_digstore(&digstore_path, auto_yes)
64                .map_err(|e| DigstoreError::internal(format!("Interactive prompt failed: {}", e)))?
65            {
66                println!();
67                println!("{}", "Initialization cancelled".yellow());
68                return Err(DigstoreError::store_already_exists(
69                    project_path.to_path_buf(),
70                ));
71            }
72
73            // Before removing files, ensure any existing stores are properly closed
74            // This prevents Windows "user-mapped section" errors
75            if let Ok(existing_digstore) = crate::core::digstore_file::DigstoreFile::load(&digstore_path) {
76                if let Ok(existing_store_id) = existing_digstore.get_store_id() {
77                    let archive_path = get_archive_path(&existing_store_id)?;
78                    let staging_path = archive_path.with_extension("staging.bin");
79                    
80                    // On Windows, don't try to access the files - just remove them directly
81                    // after ensuring any handles are released
82                    #[cfg(windows)]
83                    {
84                        // Force garbage collection to release any potential file handles
85                        for _ in 0..3 {
86                            std::hint::black_box(());
87                            std::thread::sleep(std::time::Duration::from_millis(100));
88                        }
89                        
90                        // Remove old archive files directly without opening them
91                        for path in [&archive_path, &staging_path] {
92                            if path.exists() {
93                                let _ = std::fs::remove_file(path);
94                            }
95                        }
96                    }
97                    
98                    #[cfg(not(windows))]
99                    {
100                        // On non-Windows, try to sync files before removal
101                        for path in [&archive_path, &staging_path] {
102                            if path.exists() {
103                                let _ = std::fs::OpenOptions::new()
104                                    .read(true)
105                                    .open(path)
106                                    .and_then(|file| file.sync_all());
107                                let _ = std::fs::remove_file(path);
108                            }
109                        }
110                    }
111                }
112            }
113
114            // On Windows, use longer delay and force garbage collection
115            #[cfg(windows)]
116            {
117                std::thread::sleep(std::time::Duration::from_millis(500));
118                // Force garbage collection multiple times to ensure cleanup
119                for _ in 0..3 {
120                    std::hint::black_box(());
121                    std::thread::sleep(std::time::Duration::from_millis(50));
122                }
123            }
124            
125            #[cfg(not(windows))]
126            std::thread::sleep(std::time::Duration::from_millis(100));
127
128            // Remove existing .digstore file to proceed with new initialization
129            std::fs::remove_file(&digstore_path)?;
130            println!();
131            println!("{}", "Existing repository file removed".yellow());
132        }
133
134        // Generate new store ID
135        let store_id = generate_store_id();
136
137        // Get archive path
138        let archive_path = get_archive_path(&store_id)?;
139
140        // Create parent directory for .dig files
141        if let Some(parent) = archive_path.parent() {
142            std::fs::create_dir_all(parent)?;
143        }
144
145        // Create .digstore file
146        let repository_name = project_path
147            .file_name()
148            .and_then(|n| n.to_str())
149            .map(|s| s.to_string());
150
151        let digstore_file = DigstoreFile::new(store_id, repository_name);
152        digstore_file.save(&digstore_path)?;
153
154        // Create archive with optional encryption
155        let dig_archive = DigArchive::create(archive_path.clone())?;
156        let archive = EncryptedArchive::new_with_store_id(dig_archive, Some(&store_id))?;
157
158        // Initialize binary staging area (in same directory as archive)
159        let staging_path = archive_path.with_extension("staging.bin");
160        let mut staging = BinaryStagingArea::new(staging_path);
161        staging.initialize()?;
162
163        // Initialize Layer 0 (metadata layer)
164        let mut store = Self {
165            store_id,
166            archive,
167            project_path: Some(project_path.to_path_buf()),
168            current_root: None,
169            staging,
170            chunking_engine: ChunkingEngine::new(),
171            streaming_engine: StreamingChunkingEngine::new(),
172            batch_processor: BatchProcessor::new(),
173        };
174
175        store.init_layer_zero()?;
176
177        Ok(store)
178    }

185    /// Open an existing store with options
186    pub fn open_with_options(project_path: &Path, auto_yes: bool) -> Result<Self> {
187        let digstore_path = project_path.join(".digstore");
188        if !digstore_path.exists() {
189            // No .digstore file found - provide helpful guidance
190            eprintln!("{}", "No repository found!".red().bold());
191            eprintln!(
192                "  Looking for: {}",
193                digstore_path.display().to_string().yellow()
194            );
195            eprintln!();
196            eprintln!("{}", "This directory is not a Digstore repository.".blue());
197            eprintln!();
198            eprintln!("{}", "To create a new repository:".green().bold());
199            eprintln!("  {}", "digstore init --name \"my-project\"".cyan());
200            eprintln!();
201
202            return Err(DigstoreError::store_not_found(project_path.to_path_buf()));
203        }
204
205        let mut digstore_file = DigstoreFile::load(&digstore_path)?;
206        let store_id = digstore_file.get_store_id()?;
207        let archive_path = get_archive_path(&store_id)?;
208
209        // Check for migration from old directory format
210        let old_global_path = get_global_store_path(&store_id)?;
211        let archive = if archive_path.exists() {
212            // Open existing archive with optional encryption
213            let dig_archive = DigArchive::open(archive_path.clone())?;
214            EncryptedArchive::new_with_store_id(dig_archive, Some(&store_id))?
215        } else if old_global_path.exists() {
216            // Migrate from old directory format
217            println!("Migrating store from directory to archive format...");
218            let dig_archive =
219                DigArchive::migrate_from_directory(archive_path.clone(), &old_global_path)?;
220
221            // Clean up old directory after successful migration
222            std::fs::remove_dir_all(&old_global_path)?;
223            let archive = EncryptedArchive::new_with_store_id(dig_archive, Some(&store_id))?;
224            println!("✓ Migration completed successfully");
225            archive
226        } else {
227            // Store archive not found - offer interactive recreation
228            use crate::cli::interactive;
229
230            match interactive::handle_missing_store(
231                &archive_path,
232                &store_id.to_hex(),
233                project_path,
234                auto_yes,
235            )
236            .map_err(|e| DigstoreError::internal(format!("Interactive prompt failed: {}", e)))
237            {
238                Ok(new_store) => return Ok(new_store),
239                Err(e) => return Err(e),
240            }
241        };
242
243        // Update last accessed time
244        digstore_file.update_last_accessed();
245        digstore_file.save(&digstore_path)?;
246
247        // Load current root hash from Layer 0 in archive
248        let current_root = Self::load_current_root_from_archive(&archive)?;
249
250        // Load persistent binary staging
251        let staging_path = archive_path.with_extension("staging.bin");
252        let mut staging = BinaryStagingArea::new(staging_path);
253        staging.load()?;
254
255        Ok(Self {
256            store_id,
257            archive,
258            project_path: Some(project_path.to_path_buf()),
259            current_root,
260            staging,
261            chunking_engine: ChunkingEngine::new(),
262            streaming_engine: StreamingChunkingEngine::new(),
263            batch_processor: BatchProcessor::new(),
264        })
265    }

315    /// Add a single file to staging (internal method) with smart change detection
316    fn add_file_internal(&mut self, file_path: &Path) -> Result<()> {
317        // Resolve relative to project directory if available
318        let full_path = if let Some(project_path) = &self.project_path {
319            if file_path.is_relative() {
320                project_path.join(file_path)
321            } else {
322                file_path.to_path_buf()
323            }
324        } else {
325            file_path.to_path_buf()
326        };
327
328        // Check if file exists
329        if !full_path.exists() {
330            return Err(DigstoreError::file_not_found(full_path));
331        }
332
333        if !full_path.is_file() {
334            return Err(DigstoreError::invalid_file_path(full_path));
335        }
336
337        // Smart staging: Check if file has changed since last commit
338        if let Some(current_root) = self.current_root {
339            if let Ok(committed_hash) = self.get_committed_file_hash(file_path, current_root) {
340                // Compute current file hash to compare
341                let current_hash = crate::core::hash::hash_file(&full_path)?;
342
343                if current_hash == committed_hash {
344                    // File hasn't changed - skip staging
345                    return Ok(());
346                }
347            }
348        }
349
350        // Get file size to determine processing strategy
351        let file_size = std::fs::metadata(&full_path)?.len();
352
353        // Use streaming processing - NEVER load entire file into memory
354        let chunks = if file_size > 10 * 1024 * 1024 {
355            // Large files: use streaming chunking engine
356            self.streaming_engine.chunk_file_streaming(&full_path)?
357        } else {
358            // Smaller files: use regular chunking but still streaming
359            self.chunking_engine.chunk_file_streaming(&full_path)?
360        };
361
362        // Create file entry from chunk metadata (no file data stored)
363        let file_hash = Self::compute_file_hash_from_chunks(&chunks);
364        let file_entry = crate::core::types::FileEntry {
365            path: file_path.to_path_buf(),
366            hash: file_hash,
367            size: file_size,
368            chunks: chunks
369                .iter()
370                .map(|c| crate::core::types::ChunkRef {
371                    hash: c.hash,
372                    offset: c.offset,
373                    size: c.size,
374                })
375                .collect(),
376            metadata: FileMetadata {
377                mode: 0o644,
378                modified: std::fs::metadata(&full_path)?
379                    .modified()
380                    .ok()
381                    .and_then(|t| t.duration_since(std::time::UNIX_EPOCH).ok())
382                    .map(|d| d.as_secs() as i64)
383                    .unwrap_or(0),
384                is_new: true,
385                is_modified: false,
386                is_deleted: false,
387            },
388        };
389
390        // Convert to binary staged file format
391        let binary_staged_file = BinaryStagedFile {
392            path: file_path.to_path_buf(),
393            hash: file_hash,
394            size: file_size,
395            chunks,
396            modified_time: std::fs::metadata(&full_path)?.modified().ok(),
397        };
398
399        // Add to binary staging area
400        self.staging.stage_file_streaming(binary_staged_file)?;
401
402        Ok(())
403    }

412    /// Add many files efficiently using batch processing
413    pub fn add_files_batch(
414        &mut self,
415        files: Vec<PathBuf>,
416        progress: Option<&indicatif::ProgressBar>,
417    ) -> Result<()> {
418        if files.is_empty() {
419            return Ok(());
420        }
421
422        // Convert absolute paths to relative paths for storage, but keep absolute for processing
423        let project_root = self
424            .project_path
425            .as_ref()
426            .cloned()
427            .unwrap_or_else(|| std::env::current_dir().unwrap_or_default());
428        let files_with_relative: Vec<(PathBuf, PathBuf)> = files
429            .into_iter()
430            .map(|abs_path| {
431                // Try to make path relative to project root
432                let relative_path = if let Ok(rel_path) = abs_path.strip_prefix(&project_root) {
433                    rel_path.to_path_buf()
434                } else {
435                    // Fallback: use the absolute path as-is for now
436                    abs_path.clone()
437                };
438                (abs_path, relative_path)
439            })
440            .collect();
441
442        // Use batch processing for efficiency (with absolute paths)
443        let absolute_paths: Vec<PathBuf> = files_with_relative
444            .iter()
445            .map(|(abs, _)| abs.clone())
446            .collect();
447        let batch_result = self
448            .batch_processor
449            .process_files_batch(absolute_paths, progress)?;
450
451        // Store metrics before consuming batch_result
452        let file_count = batch_result.file_entries.len();
453        let metrics = batch_result.performance_metrics.clone();
454        let dedup_stats = batch_result.deduplication_stats.clone();
455
456        // Convert to binary staged files and add in batch
457        let mut binary_staged_files = Vec::with_capacity(batch_result.file_entries.len());
458
459        for (i, (mut file_entry, file_chunks)) in batch_result
460            .file_entries
461            .into_iter()
462            .zip(batch_result.chunks.into_iter())
463            .enumerate()
464        {
465            // Update file entry to use relative path
466            if let Some((abs_path, relative_path)) = files_with_relative.get(i) {
467                file_entry.path = relative_path.clone();
468
469                // Get file metadata
470                let modified_time = std::fs::metadata(abs_path)?.modified().ok();
471
472                binary_staged_files.push(BinaryStagedFile {
473                    path: relative_path.clone(),
474                    hash: file_entry.hash,
475                    size: file_entry.size,
476                    chunks: vec![file_chunks], // Store the actual chunks
477                    modified_time,
478                });
479            }
480        }
481
482        // Add all files to binary staging in one batch operation
483        self.staging.stage_files_batch(binary_staged_files)?;
484
485        // Print performance summary
486        println!(
487            "  • Processed {} files at {:.1} files/s ({:.1} MB/s)",
488            file_count, metrics.files_per_second, metrics.mb_per_second
489        );
490
491        if dedup_stats.deduplication_ratio > 0.01 {
492            println!(
493                "  • Deduplication: {:.1}% ({} bytes saved)",
494                dedup_stats.deduplication_ratio * 100.0,
495                dedup_stats.bytes_saved
496            );
497        }
498
499        Ok(())
500    }

502    /// Add a directory recursively
503    pub fn add_directory(&mut self, dir_path: &Path, recursive: bool) -> Result<()> {
504        if !dir_path.exists() {
505            return Err(DigstoreError::file_not_found(dir_path.to_path_buf()));
506        }
507
508        if !dir_path.is_dir() {
509            return Err(DigstoreError::invalid_file_path(dir_path.to_path_buf()));
510        }
511
512        if recursive {
513            use walkdir::WalkDir;
514
515            // Collect all files first (use absolute paths for batch processing)
516            let mut files = Vec::new();
517            for entry in WalkDir::new(dir_path)
518                .follow_links(false)
519                .into_iter()
520                .filter_map(|e| e.ok())
521                .filter(|e| e.file_type().is_file())
522            {
523                // Store absolute path for batch processing
524                files.push(entry.path().to_path_buf());
525            }
526
527            // Use batch processing if many files, otherwise process individually
528            if files.len() > 10 {
529                println!("  • Using batch processing for {} files", files.len());
530                self.add_files_batch(files, None)?;
531                return Ok(());
532            } else {
533                // Process individually for small numbers of files
534                for file_path in files {
535                    self.add_file_internal(&file_path)?;
536                }
537            }
538        } else {
539            // Add only direct files in directory
540            for entry in std::fs::read_dir(dir_path)? {
541                let entry = entry?;
542                let path = entry.path();
543
544                if path.is_file() {
545                    let relative_path = path
546                        .strip_prefix(
547                            self.project_path
548                                .as_ref()
549                                .unwrap_or(&std::env::current_dir()?),
550                        )
551                        .unwrap_or(&path);
552
553                    self.add_file_internal(relative_path)?;
554                }
555            }
556        }
557
558        // Save staging to disk
559        self.staging.flush()?;
560
561        Ok(())
562    }

564    /// Create a commit from staged files (cumulative approach for MVP)
565    pub fn commit(&mut self, message: &str) -> Result<LayerId> {
566        if self.staging.staged_count() == 0 {
567            return Err(DigstoreError::internal("No files staged for commit"));
568        }
569
570        // Create new layer
571        let layer_number = self.get_next_layer_number()?;
572        let parent_hash = self.current_root.unwrap_or(Hash::zero());
573        let mut layer = Layer::new(LayerType::Full, layer_number, parent_hash);
574
575        // For MVP: Create cumulative layers that include all files from previous commits
576        // First, add all files from the previous layer (if any)
577        if let Some(current_root) = self.current_root {
578            if let Ok(previous_layer) = self.load_layer(current_root) {
579                for file_entry in previous_layer.files {
580                    // Only add if not being replaced by staged version
581                    if !self.staging.is_staged(&file_entry.path) {
582                        layer.add_file(file_entry.clone());
583
584                        // Add chunks for this file
585                        for chunk in &previous_layer.chunks {
586                            // Check if this chunk belongs to this file
587                            for chunk_ref in &file_entry.chunks {
588                                if chunk_ref.hash == chunk.hash {
589                                    // Ensure chunk has data when copying from previous layer
590                                    layer.add_chunk(chunk.clone());
591                                    break;
592                                }
593                            }
594                        }
595                    }
596                }
597            }
598        }
599
600        // Add all staged files and chunks to layer
601        let staged_files = self.staging.get_all_staged_files()?;
602        for staged_file in &staged_files {
603            // Convert BinaryStagedFile to FileEntry
604            let file_entry = FileEntry {
605                path: staged_file.path.clone(),
606                hash: staged_file.hash,
607                size: staged_file.size,
608                chunks: staged_file
609                    .chunks
610                    .iter()
611                    .map(|c| ChunkRef {
612                        hash: c.hash,
613                        offset: c.offset,
614                        size: c.size,
615                    })
616                    .collect(),
617                metadata: FileMetadata {
618                    mode: 0o644,
619                    modified: staged_file
620                        .modified_time
621                        .and_then(|t| t.duration_since(std::time::UNIX_EPOCH).ok())
622                        .map(|d| d.as_secs() as i64)
623                        .unwrap_or(0),
624                    is_new: true,
625                    is_modified: false,
626                    is_deleted: false,
627                },
628            };
629
630            layer.add_file(file_entry);
631
632            // Read actual chunk data from the file
633            let full_path = if let Some(project_path) = &self.project_path {
634                if staged_file.path.is_relative() {
635                    project_path.join(&staged_file.path)
636                } else {
637                    staged_file.path.clone()
638                }
639            } else {
640                staged_file.path.clone()
641            };
642
643            if full_path.exists() {
644                use std::fs::File;
645                use std::io::{Read, Seek, SeekFrom};
646
647                let mut file = File::open(&full_path)?;
648
649                for chunk in &staged_file.chunks {
650                    // Read chunk data from the file at the specified offset
651                    file.seek(SeekFrom::Start(chunk.offset))?;
652                    let mut chunk_data = vec![0u8; chunk.size as usize];
653                    file.read_exact(&mut chunk_data)?;
654
655                    // Check if the archive is configured for encryption
656                    let final_data = if self.archive.is_encrypted() {
657                        // Check if we're using custom encryption (store-level secret)
658                        if let Some(custom_key) = self.archive.get_encryption_key() {
659                            // Use custom encryption key for truly secret data (store-level)
660                            crate::crypto::encrypt_data_with_key(&chunk_data, &custom_key)?
661                        } else {
662                            // Create URN for this chunk (use store ID and chunk hash)
663                            let chunk_urn = format!(
664                                "urn:dig:chia:{}/chunk/{}",
665                                self.store_id.to_hex(),
666                                chunk.hash.to_hex()
667                            );
668
669                            // Encrypt chunk data using URN (wallet-based encryption)
670                            crate::crypto::encrypt_data(&chunk_data, &chunk_urn)?
671                        }
672                    } else {
673                        chunk_data
674                    };
675
676                    // Create chunk with actual data
677                    let chunk_with_data = Chunk {
678                        hash: chunk.hash,
679                        offset: chunk.offset,
680                        size: chunk.size,
681                        data: final_data,
682                    };
683
684                    layer.add_chunk(chunk_with_data);
685                }
686            } else {
687                return Err(DigstoreError::file_not_found(full_path));
688            }
689        }
690
691        // Set commit message in metadata
692        layer.metadata.message = Some(message.to_string());
693
694        // Get author from global config
695        let mut global_config = crate::config::GlobalConfig::load()?;
696        global_config.ensure_user_configured()?;
697
698        let author_name = global_config.get_author_name();
699        let author_email = global_config.get_author_email();
700
701        let author_string = if let Some(email) = author_email {
702            format!("{} <{}>", author_name, email)
703        } else {
704            author_name
705        };
706
707        layer.metadata.author = Some(author_string);
708
709        // Compute layer ID
710        let layer_id = layer.compute_layer_id()?;
711        layer.metadata.layer_id = layer_id;
712
713        // Layer will be stored inside the archive file (not as separate .layer file)
714
715        // Serialize layer to bytes
716        let layer_data = layer.serialize_to_bytes()?;
717
718        // Add layer to archive
719        self.archive.add_layer(layer_id, &layer_data)?;
720
721        // Update root history in Layer 0
722        self.update_root_history(layer_id)?;
723
724        // Update current root
725        self.current_root = Some(layer_id);
726
727        // Close memory maps before clearing (Windows file locking fix)
728        self.staging.mmap = None;
729        self.staging.mmap_mut = None;
730
731        // Clear binary staging
732        self.staging.clear()?;
733
734        Ok(layer_id)
735    }

743    /// Get a file by path from the latest commit
744    pub fn get_file(&self, file_path: &Path) -> Result<Vec<u8>> {
745        // First check if file is in staging
746        if let Some(staged_file) = self.staging.get_staged_file(file_path)? {
747            // For binary staging, we need to reconstruct the file from chunks
748            // Read the actual file from disk since staging only stores metadata
749            let full_path = if let Some(project_path) = &self.project_path {
750                if staged_file.path.is_relative() {
751                    project_path.join(&staged_file.path)
752                } else {

754                }
755            } else {
756                staged_file.path.clone()
757            };
758
759            if full_path.exists() {
760                return Ok(std::fs::read(full_path)?);
761            } else {
762                return Err(DigstoreError::file_not_found(full_path));
763            }
764        }
765
766        // Get from committed layers
767        self.get_file_at(file_path, self.current_root)

770    /// Get a file at a specific root hash (returns encrypted data as-is)
771    pub fn get_file_at(&self, file_path: &Path, root_hash: Option<RootHash>) -> Result<Vec<u8>> {
772        let target_root = root_hash.unwrap_or(
773            self.current_root
774                .ok_or_else(|| DigstoreError::file_not_found(file_path.to_path_buf()))?,
775        );
776
777        // Load layer from archive
778        let layer = self.archive.get_layer(&target_root)?;
779
780        // Find file in layer
781        for file_entry in &layer.files {
782            if file_entry.path == file_path {
783                // Reconstruct file from chunks (return encrypted data as-is)
784                let mut file_chunks = Vec::new();
785
786                for chunk_ref in &file_entry.chunks {
787                    // Find chunk in layer
788                    for chunk in &layer.chunks {
789                        if chunk.hash == chunk_ref.hash {
790                            // Return encrypted chunk data as-is - no decryption here
791                            file_chunks.push(chunk.clone());
792                            break;
793                        }
794                    }
795                }
796
797                return Ok(self.chunking_engine.reconstruct_from_chunks(&file_chunks));
798            }
799        }
800
801        Err(DigstoreError::file_not_found(file_path.to_path_buf()))
802    }

902    /// Load current root hash from Layer 0 in archive
903    fn load_current_root_from_archive(archive: &EncryptedArchive) -> Result<Option<RootHash>> {
904        let layer_zero_hash = Hash::zero();
905
906        if !archive.has_layer(&layer_zero_hash) {
907            return Ok(None);
908        }
909
910        // Get Layer 0 metadata
911        let metadata_bytes = archive.get_layer_data(&layer_zero_hash)?;
912        
913        // Debug: Check if metadata is empty
914        if metadata_bytes.is_empty() {
915            return Err(DigstoreError::internal("Layer 0 metadata is empty"));
916        }
917        
918        let metadata: serde_json::Value = serde_json::from_slice(&metadata_bytes)
919            .map_err(|e| DigstoreError::internal(format!("Failed to parse Layer 0 metadata: {}. Data length: {}", e, metadata_bytes.len())))?;
920
921        if let Some(root_history) = metadata.get("root_history").and_then(|v| v.as_array()) {
922            if let Some(latest_root) = root_history.last() {
923                if let Some(root_str) = latest_root.get("root_hash").and_then(|v| v.as_str()) {
924                    if let Ok(root_hash) = Hash::from_hex(root_str) {
925                        return Ok(Some(root_hash));
926                    }
927                }
928            }
929        }
930
931        Ok(None)
932    }

978        // Load the layer for this commit

980
981        // Find the file in the layer
982        for file_entry in &layer.files {
983            if file_entry.path == file_path {
984                return Ok(file_entry.hash);

986        }

992    /// Check if a file has changed since the last commit
993    pub fn has_file_changed(&self, file_path: &Path) -> Result<bool> {
994        // If no current root, file is new
995        let current_root = match self.current_root {
996            Some(root) => root,
997            None => return Ok(true), // No commits yet, so file is new
998        };
999
1000        // Get current file hash
1001        let full_path = if let Some(project_path) = &self.project_path {
1002            if file_path.is_relative() {
1003                project_path.join(file_path)
1004            } else {
1005                file_path.to_path_buf()
1006            }
1007        } else {
1008            file_path.to_path_buf()
1009        };
1010
1011        if !full_path.exists() {
1012            return Err(DigstoreError::file_not_found(full_path));
1013        }
1014
1015        let current_hash = crate::core::hash::hash_file(&full_path)?;
1016
1017        // Compare with committed hash
1018        match self.get_committed_file_hash(file_path, current_root) {
1019            Ok(committed_hash) => Ok(current_hash != committed_hash),
1020            Err(_) => Ok(true), // File not in last commit, so it's new/changed
1021        }
1022    }

1025/// Generate a new random store ID
1026pub fn generate_store_id() -> StoreId {
1027    let mut bytes = [0u8; 32];
1028    getrandom::getrandom(&mut bytes).unwrap_or_else(|_| {
1029        // Fallback to system time + process ID if getrandom fails
1030        use std::time::{SystemTime, UNIX_EPOCH};
1031        let timestamp = SystemTime::now()
1032            .duration_since(UNIX_EPOCH)
1033            .unwrap_or_default()
1034            .as_nanos() as u64;
1035        let pid = std::process::id() as u64;
1036        let combined = timestamp.wrapping_mul(pid);
1037
1038        for (i, chunk) in combined.to_le_bytes().iter().enumerate() {
1039            if i < 32 {
1040                bytes[i] = *chunk;
1041            }
1042        }
1043
1044        // Fill remaining bytes with a simple pattern
1045        for i in 8..32 {
1046            bytes[i] = ((i * 7) % 256) as u8;
1047        }
1048    });
1049    Hash::from_bytes(bytes)
1050}
