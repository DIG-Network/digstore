113    /// Read layer from content bytes
114    fn read_from_content(content: &[u8]) -> Result<Self> {
115        let layer_data: serde_json::Value = serde_json::from_slice(content)?;
116
117        // Parse header
118        let header_data = layer_data
119            .get("header")
120            .ok_or_else(|| DigstoreError::invalid_layer_format("Missing header section"))?;
121
122        let layer_type_str = header_data
123            .get("layer_type")
124            .and_then(|v| v.as_str())
125            .ok_or_else(|| DigstoreError::invalid_layer_format("Missing layer_type"))?;
126
127        let layer_type = match layer_type_str {
128            "Header" => LayerType::Header,
129            "Full" => LayerType::Full,
130            "Delta" => LayerType::Delta,
131            _ => return Err(DigstoreError::invalid_layer_format("Invalid layer_type")),
132        };
133
134        let layer_number = header_data
135            .get("layer_number")
136            .and_then(|v| v.as_u64())
137            .ok_or_else(|| DigstoreError::invalid_layer_format("Missing layer_number"))?;
138
139        let parent_hash_str = header_data
140            .get("parent_hash")
141            .and_then(|v| v.as_str())
142            .ok_or_else(|| DigstoreError::invalid_layer_format("Missing parent_hash"))?;
143
144        let parent_hash = Hash::from_hex(parent_hash_str)
145            .map_err(|_| DigstoreError::invalid_layer_format("Invalid parent_hash"))?;
146
147        // Create header
148        let mut header = LayerHeader::new(layer_type, layer_number, parent_hash);
149
150        // Parse metadata
151        let metadata: LayerMetadata = serde_json::from_value(
152            layer_data
153                .get("metadata")
154                .cloned()
155                .ok_or_else(|| DigstoreError::invalid_layer_format("Missing metadata section"))?,
156        )?;
157
158        // Parse files
159        let files: Vec<FileEntry> = serde_json::from_value(
160            layer_data
161                .get("files")
162                .cloned()
163                .ok_or_else(|| DigstoreError::invalid_layer_format("Missing files section"))?,
164        )?;
165
166        // Parse chunks
167        let chunks: Vec<Chunk> = serde_json::from_value(
168            layer_data
169                .get("chunks")
170                .cloned()
171                .ok_or_else(|| DigstoreError::invalid_layer_format("Missing chunks section"))?,
172        )?;
173
174        // Update header counts to match actual data
175        header.files_count = files.len() as u32;
176        header.chunks_count = chunks.len() as u32;
177
178        Ok(Self {
179            header,
180            metadata,
181            files,
182            chunks,
183        })
184    }

186    /// Verify layer integrity
187    pub fn verify(&self) -> Result<bool> {
188        // Verify header is valid
189        if !self.header.is_valid() {
190            return Ok(false);
191        }
192
193        // For empty layers, don't check counts (they're set during serialization)
194        if !self.files.is_empty() || !self.chunks.is_empty() {
195            // Verify counts match
196            if self.header.files_count != self.files.len() as u32 {
197                return Ok(false);
198            }
199
200            if self.header.chunks_count != self.chunks.len() as u32 {
201                return Ok(false);
202            }
203        }
204
205        // Verify chunk hashes
206        for chunk in &self.chunks {
207            let computed_hash = crate::core::hash::sha256(&chunk.data);
208            if computed_hash != chunk.hash {
209                return Ok(false);
210            }
211        }
212
213        // Verify file hashes
214        for file_entry in &self.files {
215            // Find chunks for this file and verify file hash
216            let mut file_chunks = Vec::new();
217            for chunk_ref in &file_entry.chunks {
218                if let Some(chunk) = self.chunks.iter().find(|c| c.hash == chunk_ref.hash) {
219                    file_chunks.push(chunk.clone());
220                }
221            }
222
223            // Reconstruct file data and verify hash
224            let mut file_data = Vec::new();
225            for chunk in file_chunks {
226                file_data.extend_from_slice(&chunk.data);
227            }
228
229            let computed_file_hash = crate::core::hash::sha256(&file_data);
230            if computed_file_hash != file_entry.hash {
231                return Ok(false);
232            }
233        }
234
235        Ok(true)
236    }

238    /// Serialize the index section
239    fn serialize_index(&self) -> Result<Vec<u8>> {
240        let mut index_data = Vec::new();
241
242        // Index header (6 bytes)
243        index_data.extend_from_slice(&1u16.to_le_bytes()); // Version
244        index_data
245            .extend_from_slice(&((self.files.len() + self.chunks.len()) as u32).to_le_bytes()); // Total entries
246
247        // File entries
248        for file in &self.files {
249            let path_str = file.path.to_string_lossy();
250            let path_bytes = path_str.as_bytes();
251
252            // Path length (2 bytes)
253            index_data.extend_from_slice(&(path_bytes.len() as u16).to_le_bytes());
254
255            // Path (variable)
256            index_data.extend_from_slice(path_bytes);
257
258            // File size (8 bytes)
259            index_data.extend_from_slice(&file.size.to_le_bytes());
260
261            // File hash (32 bytes)
262            index_data.extend_from_slice(file.hash.as_bytes());
263
264            // Chunk count (2 bytes)
265            index_data.extend_from_slice(&(file.chunks.len() as u16).to_le_bytes());
266
267            // First chunk index (4 bytes) - simplified for now
268            index_data.extend_from_slice(&0u32.to_le_bytes());
269
270            // Metadata (JSON serialized)
271            let metadata_json = serde_json::to_vec(&file.metadata)?;
272            index_data.extend_from_slice(&(metadata_json.len() as u16).to_le_bytes());
273            index_data.extend_from_slice(&metadata_json);
274        }
275
276        // Chunk entries
277        for chunk in &self.chunks {
278            // Chunk hash (32 bytes)
279            index_data.extend_from_slice(chunk.hash.as_bytes());
280
281            // Offset in file (8 bytes)
282            index_data.extend_from_slice(&chunk.offset.to_le_bytes());
283
284            // Chunk size (4 bytes)
285            index_data.extend_from_slice(&chunk.size.to_le_bytes());
286
287            // Data offset in layer (8 bytes) - will be calculated
288            index_data.extend_from_slice(&0u64.to_le_bytes());
289
290            // Compressed size (4 bytes) - same as size for now (no compression)
291            index_data.extend_from_slice(&chunk.size.to_le_bytes());
292
293            // Flags (1 byte)
294            index_data.push(0);
295        }
296
297        Ok(index_data)
298    }

300    /// Deserialize the index section
301    fn deserialize_index(index_data: &[u8]) -> Result<(Vec<FileEntry>, Vec<Chunk>)> {
302        let mut offset = 0;
303
304        // Read index header
305        if index_data.len() < 6 {
306            return Err(DigstoreError::invalid_layer_format("Index too short"));
307        }
308
309        let _version = u16::from_le_bytes([index_data[offset], index_data[offset + 1]]);
310        offset += 2;
311
312        let _entries_count = u32::from_le_bytes([
313            index_data[offset],
314            index_data[offset + 1],
315            index_data[offset + 2],
316            index_data[offset + 3],
317        ]);
318        offset += 4;
319
320        // For now, return empty vectors (will implement proper parsing later)
321        // This is a simplified implementation - full parsing would be complex
322        let files = Vec::new();
323        let chunks = Vec::new();
324
325        Ok((files, chunks))
326    }

328    /// Serialize the data section
329    fn serialize_data(&self) -> Result<Vec<u8>> {
330        let mut data_section = Vec::new();
331
332        // For each chunk, write size + data
333        for chunk in &self.chunks {
334            // 4-byte size prefix
335            data_section.extend_from_slice(&chunk.size.to_le_bytes());
336
337            // Chunk data
338            data_section.extend_from_slice(&chunk.data);
339        }
340
341        Ok(data_section)
342    }

344    /// Serialize the merkle tree section
345    fn serialize_merkle(&self) -> Result<Vec<u8>> {
346        let mut merkle_data = Vec::new();
347
348        if self.files.is_empty() {
349            return Ok(merkle_data);
350        }
351
352        // Calculate tree depth
353        let leaf_count = self.files.len();
354        let depth = if leaf_count <= 1 {
355            0
356        } else {
357            (leaf_count as f64).log2().ceil() as u8
358        };
359
360        // Tree header
361        merkle_data.push(depth);
362        merkle_data.extend_from_slice(&(leaf_count as u32).to_le_bytes());
363
364        // Include file hashes as leaves
365        for file in &self.files {
366            merkle_data.extend_from_slice(file.hash.as_bytes());
367        }
368
369        // Build and include merkle tree
370        if leaf_count > 1 {
371            let file_hashes: Vec<_> = self.files.iter().map(|f| f.hash).collect();
372            let merkle_tree = crate::proofs::merkle::MerkleTree::from_hashes(&file_hashes)?;
373            merkle_data.extend_from_slice(merkle_tree.root().as_bytes());
374        }
375
376        Ok(merkle_data)
377    }
