16/// Execute the proof generate-archive-size command
17pub fn execute(
18    store_id: Option<String>,
19    output: Option<PathBuf>,
20    format: Option<String>,
21    verbose: bool,
22    show_compression: bool,
23    json: bool,
24) -> Result<()> {
25    println!(
26        "{}",
27        "Generating tamper-proof archive size proof...".bright_blue()
28    );
29
30    // Determine store ID (auto-detect from .digstore file or use provided)
31    let (store_id_string, store_id_hash) = match store_id {
32        Some(provided_store_id) => {
33            if verbose {
34                println!(
35                    "  {} Using provided store ID: {}",
36                    "•".cyan(),
37                    provided_store_id
38                );
39            }
40            let parsed_hash = Hash::from_hex(&provided_store_id)
41                .map_err(|_| anyhow::anyhow!("Invalid store ID format: {}", provided_store_id))?;
42            (provided_store_id, parsed_hash)
43        },
44        None => {
45            if verbose {
46                println!(
47                    "  {} Auto-detecting store ID from .digstore file...",
48                    "•".cyan()
49                );
50            }
51
52            // Find .digstore file in current directory or parent directories
53            let repo_root = find_repository_root()
54                .map_err(|e| anyhow::anyhow!("Failed to search for repository: {}", e))?
55                .ok_or_else(|| anyhow::anyhow!(
56                    "No .digstore file found in current directory or parent directories.\n  \
57                     Either provide store ID as argument or run from a digstore repository directory."
58                ))?;
59
60            // Load .digstore file to get store ID
61            let digstore_file_path = repo_root.join(".digstore");
62            let digstore_file = DigstoreFile::load(&digstore_file_path)
63                .map_err(|e| anyhow::anyhow!("Failed to load .digstore file: {}", e))?;
64
65            let detected_store_id = digstore_file.store_id.clone();
66            let parsed_hash = digstore_file
67                .get_store_id()
68                .map_err(|e| anyhow::anyhow!("Invalid store ID in .digstore file: {}", e))?;
69
70            if verbose {
71                println!("  {} Detected store ID: {}", "✓".green(), detected_store_id);
72                if let Some(repo_name) = &digstore_file.repository_name {
73                    println!("  {} Repository: {}", "•".cyan(), repo_name);
74                }
75            }
76
77            (detected_store_id, parsed_hash)
78        },
79    };
80
81    if verbose {
82        println!(
83            "  {} Auto-discovering parameters from .dig directory...",
84            "•".cyan()
85        );
86    }
87
88    // Get publisher's public key from specified wallet profile or active wallet
89    let wallet_profile = CliContext::get_wallet_profile();
90    let wallet_public_key = WalletManager::get_wallet_public_key(wallet_profile).map_err(|e| {
91        anyhow::anyhow!("Failed to get public key from wallet: {}. Ensure you have a wallet initialized with 'digstore wallet create <profile>' or 'digstore wallet list' to see available wallets.", e)
92    })?;
93
94    let publisher_public_key = wallet_public_key.to_hex();
95
96    if verbose {
97        println!(
98            "  {} Publisher public key: {}...",
99            "•".cyan(),
100            &publisher_public_key[..16]
101        );
102        println!("  {} Final store ID: {}", "•".cyan(), store_id_string);
103    }
104
105    // Open the store to get current root hash and archive size
106    let store = Store::open_global(&store_id_hash).map_err(|e| {
107        anyhow::anyhow!(
108            "Failed to open store {}: {}. Ensure the store exists in ~/.dig/",
109            store_id_string,
110            e
111        )
112    })?;
113
114    let current_root_hash = store.current_root.ok_or_else(|| {
115        anyhow::anyhow!(
116            "Store {} has no commits yet. Please commit some data first.",
117            store_id_string
118        )
119    })?;
120
121    // Get the actual archive file size
122    let archive_path = get_archive_path(&store_id_hash)?;
123    let actual_file_size = std::fs::metadata(&archive_path)
124        .map_err(|e| anyhow::anyhow!("Failed to get archive file size: {}", e))?
125        .len();
126
127    if verbose {
128        println!(
129            "  {} Current root hash: {}",
130            "•".cyan(),
131            current_root_hash.to_hex()
132        );
133        println!(
134            "  {} Archive file size: {} bytes",
135            "•".cyan(),
136            actual_file_size
137        );
138        println!();
139    }
140
141    // Generate the proof with auto-discovered parameters
142    let proof =
143        match ArchiveSizeProof::generate(&store_id_hash, &current_root_hash, actual_file_size) {
144            Ok(mut proof) => {
145                // Include publisher's public key in the proof for verification
146                proof.publisher_public_key = Some(publisher_public_key.clone());
147
148                if verbose {
149                    println!("  {} Archive located and verified", "✓".green());
150                    println!(
151                        "  {} Layer count: {}",
152                        "•".cyan(),
153                        proof.verified_layer_count
154                    );
155                    println!(
156                        "  {} Calculated size: {} bytes",
157                        "•".cyan(),
158                        proof.calculated_total_size
159                    );
160                    println!("  {} Publisher key included in proof", "•".cyan());
161                }
162                proof
163            },
164            Err(e) => {
165                eprintln!("{} Failed to generate proof: {}", "✗".red(), e);
166                return Err(e.into());
167            },
168        };
169
170    // Convert to ultra-compressed format (absolute minimum size)
171    let ultra_output = proof.to_ultra_compressed().map_err(|e| {
172        anyhow::anyhow!(
173            "Ultra compression failed, falling back to standard compression: {}",
174            e
175        )
176    });
177
178    let final_output = match ultra_output {
179        Ok(ultra) => {
180            if verbose {
181                println!("  {} Using ultra-compressed format", "✓".green());
182            }
183            ultra
184        },
185        Err(_) => {
186            if verbose {
187                println!(
188                    "  {} Falling back to standard hex compression",
189                    "•".yellow()
190                );
191            }
192            proof.to_compressed_hex()?
193        },
194    };
195
196    if show_compression {
197        println!();
198        println!("{}", "Compression Statistics:".bright_yellow());
199        println!(
200            "  {} Original proof data: ~{} bytes",
201            "•".cyan(),
202            std::mem::size_of_val(&proof)
203        );
204        println!(
205            "  {} Final output: {} characters",
206            "•".cyan(),
207            final_output.len()
208        );
209        println!(
210            "  {} Compression achieved through binary encoding",
211            "•".cyan()
212        );
213        println!();
214    }
215
216    // Output the proof
217    match output {
218        Some(output_path) => {
219            // Write to file
220            if json {
221                let json_output = serde_json::to_string_pretty(&proof)?;
222                std::fs::write(&output_path, json_output)?;
223                println!(
224                    "{} Proof written to: {}",
225                    "✓".green(),
226                    output_path.display()
227                );
228            } else {
229                std::fs::write(&output_path, &final_output)?;
230                println!(
231                    "{} Ultra-compressed proof written to: {}",
232                    "✓".green(),
233                    output_path.display()
234                );
235            }
236        },
237        None => {
238            // Write to stdout
239            if json {
240                let json_output = serde_json::to_string_pretty(&proof)?;
241                println!("{}", json_output);
242            } else {
243                println!("{}", final_output);
244            }
245        },
246    }
247
248    if verbose && !json {
249        println!();
250        println!("{} Archive size proof generated successfully!", "✓".green());
251        println!(
252            "  {} Proof is tamper-proof and cryptographically secure",
253            "•".cyan()
254        );
255        println!(
256            "  {} Verifier can validate without accessing the archive file",
257            "•".cyan()
258        );
259    }
260
261    Ok(())
262}
