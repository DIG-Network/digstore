1//! Decrypt command implementation
2
3use crate::cli::commands::find_repository_root;
4use crate::core::types::Hash;
5use crate::storage::store::Store;
6use crate::urn::{parse_urn, Urn};
7use anyhow::Result;
8use colored::Colorize;
9use std::io::Write;
10use std::path::PathBuf;
11
12/// Execute the decrypt command
13pub fn execute(
14    path: String,
15    output: Option<PathBuf>,
16    urn: Option<String>,
17    decryption_key: Option<String>,
18    json: bool,
19) -> Result<()> {
20    println!("{}", "Decrypting content...".bright_blue());
21
22    // Read the encrypted file from input path
23    let encrypted_data = std::fs::read(&path)?;
24
25    println!(
26        "  {} Read {} bytes of encrypted data",
27        "•".cyan(),
28        encrypted_data.len()
29    );
30
31    // Determine the URN to use for decryption
32    let decryption_urn = if let Some(provided_urn) = urn {
33        // Use the provided URN
34        provided_urn
35    } else if path.starts_with("urn:dig:chia:") {
36        // The path itself is a URN
37        path.clone()
38    } else {
39        // Try to construct a URN from the file path
40        let repo_root = find_repository_root()?
41            .ok_or_else(|| anyhow::anyhow!("Not in a repository (no .digstore file found)"))?;
42
43        let store = Store::open(&repo_root)?;
44        let file_path = PathBuf::from(&path);
45
46        // Construct URN from store ID and file path
47        format!(
48            "urn:dig:chia:{}/{}",
49            store.store_id.to_hex(),
50            file_path.to_string_lossy().replace('\\', "/")
51        )
52    };
53
54    println!(
55        "  {} Using URN for decryption: {}",
56        "•".cyan(),
57        decryption_urn.dimmed()
58    );
59
60    // Decrypt the data using custom key if provided, otherwise use URN
61    let decrypted_data = if let Some(custom_key) = decryption_key {
62        println!("  {} Using custom decryption key", "•".cyan());
63        crate::crypto::decrypt_data_with_key(&encrypted_data, &custom_key)?
64    } else {
65        crate::crypto::decrypt_data(&encrypted_data, &decryption_urn)?
66    };
67
68    println!("  {} Decrypted {} bytes", "✓".green(), decrypted_data.len());
69
70    // Handle output
71    if let Some(output_path) = &output {
72        // Write to file (-o flag)
73        std::fs::write(output_path, &decrypted_data)?;
74
75        if json {
76            // JSON metadata about the operation
77            let output_info = serde_json::json!({
78                "action": "file_decrypted",
79                "input": path,
80                "output_file": output_path.display().to_string(),
81                "encrypted_size": encrypted_data.len(),
82                "decrypted_size": decrypted_data.len(),
83                "urn": decryption_urn,
84            });
85            println!("{}", serde_json::to_string_pretty(&output_info)?);
86        } else {
87            println!(
88                "{} Decrypted content written to: {}",
89                "✓".green().bold(),
90                output_path.display().to_string().bright_white()
91            );
92        }
93    } else {
94        // Stream to stdout (default behavior)
95        if json {
96            // JSON metadata to stderr, content to stdout
97            let output_info = serde_json::json!({
98                "action": "content_decrypted",
99                "input": path,
100                "encrypted_size": encrypted_data.len(),
101                "decrypted_size": decrypted_data.len(),
102                "urn": decryption_urn,
103            });
104            eprintln!("{}", serde_json::to_string_pretty(&output_info)?);
105        }
106
107        // Stream decrypted content to stdout
108        std::io::stdout().write_all(&decrypted_data)?;
109    }
110
111    Ok(())
112}
