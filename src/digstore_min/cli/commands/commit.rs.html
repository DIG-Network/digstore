8/// Execute the commit command
9pub fn execute(
10    message: String,
11    full_layer: bool,
12    author: Option<String>,
13    date: Option<String>,
14    edit: bool,
15    json: bool,
16) -> Result<()> {
17    // Find repository root
18    let repo_root = find_repository_root()?
19        .ok_or_else(|| anyhow::anyhow!("Not in a repository (no .digstore file found)"))?;
20
21    // Open the store
22    let mut store = Store::open(&repo_root)?;
23
24    // Check if there are staged files
25    let status = store.status();
26    if status.staged_files.is_empty() {
27        println!("{} No files staged for commit", "!".yellow());
28        println!(
29            "  {} Use 'digstore add <files>' to stage files first",
30            "→".cyan()
31        );
32        return Ok(());
33    }
34
35    println!("{}", "Creating commit...".bright_blue());
36    println!(
37        "  {} Staged files: {}",
38        "•".cyan(),
39        status.staged_files.len()
40    );
41    println!(
42        "  {} Total size: {} bytes",
43        "•".cyan(),
44        status.total_staged_size
45    );
46
47    if let Some(ref author_name) = author {
48        println!("  {} Author: {}", "•".cyan(), author_name);
49    }
50
51    if let Some(ref date_str) = date {
52        println!("  {} Date: {}", "•".cyan(), date_str);
53    }
54
55    let final_message = if edit {
56        // Open editor for commit message
57        edit_commit_message(&message)?
58    } else {
59        message
60    };
61
62    if full_layer {
63        println!("  {} Creating full layer (not delta)", "•".cyan());
64    }
65
66    // Create the commit
67    let commit_id = store.commit(&final_message)?;
68
69    println!();
70    if json {
71        // JSON output
72        let output = serde_json::json!({
73            "commit_id": commit_id.to_hex(),
74            "message": final_message,
75            "files": status.staged_files.len(),
76            "size": status.total_staged_size,
77            "archive_file": store.archive.path().display().to_string(),
78            "layer_id": commit_id.to_hex(),
79            "author": author,
80            "date": date,
81            "full_layer": full_layer
82        });
83        println!("{}", serde_json::to_string_pretty(&output)?);
84    } else {
85        println!("{} Commit created", "✓".green());
86        println!(
87            "  {} Commit ID: {}",
88            "→".cyan(),
89            commit_id.to_hex().bright_cyan()
90        );
91        println!("  {} Message: {}", "→".cyan(), final_message.bright_white());
92        println!("  {} Files: {}", "→".cyan(), status.staged_files.len());
93        println!("  {} Size: {} bytes", "→".cyan(), status.total_staged_size);
94
95        // Show archive file location (layers are stored inside the archive)
96        println!(
97            "  {} Archive file: {}",
98            "→".cyan(),
99            store.archive.path().display().to_string().dimmed()
100        );
101        println!("  {} Layer ID: {}", "→".cyan(), commit_id.to_hex().dimmed());
102    }
103
104    Ok(())
105}

107/// Open editor for commit message editing
108fn edit_commit_message(initial_message: &str) -> Result<String> {
109    use std::io::{Read, Write};
110    use std::process::Command;
111    use tempfile::NamedTempFile;
112
113    // Create temporary file with initial message
114    let mut temp_file = NamedTempFile::new()?;
115    writeln!(temp_file, "{}", initial_message)?;
116    writeln!(temp_file)?;
117    writeln!(
118        temp_file,
119        "# Please enter the commit message for your changes."
120    )?;
121    writeln!(temp_file, "# Lines starting with '#' will be ignored.")?;
122    temp_file.flush()?;
123
124    // Get editor from environment or use default
125    let editor = std::env::var("EDITOR")
126        .or_else(|_| std::env::var("VISUAL"))
127        .unwrap_or_else(|_| {
128            if cfg!(windows) {
129                "notepad".to_string()
130            } else {
131                "vi".to_string()
132            }
133        });
134
135    // Open editor
136    let status = Command::new(&editor).arg(temp_file.path()).status()?;
137
138    if !status.success() {
139        return Err(anyhow::anyhow!("Editor exited with non-zero status"));
140    }
141
142    // Read back the edited message
143    let mut edited_content = String::new();
144    let mut file = std::fs::File::open(temp_file.path())?;
145    file.read_to_string(&mut edited_content)?;
146
147    // Filter out comment lines and trim
148    let final_message = edited_content
149        .lines()
150        .filter(|line| !line.trim_start().starts_with('#'))
151        .collect::<Vec<_>>()
152        .join("\n")
153        .trim()
154        .to_string();
155
156    if final_message.is_empty() {
157        return Err(anyhow::anyhow!("Commit message cannot be empty"));
158    }
159
160    Ok(final_message)
161}
