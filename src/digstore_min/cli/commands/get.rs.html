1//! Get command implementation
2
3use anyhow::Result;
4use crate::storage::store::Store;
5use crate::cli::commands::find_repository_root;
6use crate::urn::{Urn, parse_urn};
7use crate::core::types::Hash;
8use std::path::PathBuf;
9use std::io::Write;
10use colored::Colorize;
11use base64;
12
13/// Execute the get command
14pub fn execute(
15    path: String,
16    output: Option<PathBuf>,
17    verify: bool,
18    metadata: bool,
19    at: Option<String>,
20    progress: bool,
21    json: bool,
22) -> Result<()> {
23    println!("{}", "Retrieving content...".bright_blue());
24
25    // Parse the at parameter if provided
26    let at_root = if let Some(hash_str) = at {
27        Some(Hash::from_hex(&hash_str)
28            .map_err(|_| anyhow::anyhow!("Invalid root hash: {}", hash_str))?)
29    } else {
30        None
31    };
32
33    let content = if path.starts_with("urn:dig:chia:") {
34        // Full URN provided - parse and resolve
35        println!("  {} Parsing URN: {}", "•".cyan(), path.dimmed());
36        let urn = parse_urn(&path)?;
37        
38        // For URN resolution, we need to open the store by ID
39        let store = Store::open_global(&urn.store_id)?;
40        urn.resolve(&store)?
41    } else {
42        // Simple path - find repository and resolve
43        let repo_root = find_repository_root()?
44            .ok_or_else(|| anyhow::anyhow!("Not in a repository (no .digstore file found)"))?;
45
46        let store = Store::open(&repo_root)?;
47        let file_path = PathBuf::from(&path);
48        
49        println!("  {} Retrieving file: {}", "•".cyan(), file_path.display());
50        
51        if let Some(root_hash) = at_root {
52            store.get_file_at(&file_path, Some(root_hash))?
53        } else {
54            store.get_file(&file_path)?
55        }
56    };
57
58    if progress {
59        println!("  {} Retrieved {} bytes", "✓".green(), content.len());
60    }
61
62    // Handle output
63    if let Some(output_path) = &output {
64        // Write to file (-o flag)
65        std::fs::write(output_path, &content)?;
66        
67        if json {
68            // JSON metadata about the file operation
69            let output_info = serde_json::json!({
70                "action": "file_written",
71                "path": path,
72                "output_file": output_path.display().to_string(),
73                "size": content.len(),
74                "verified": verify,
75                "at_root": at_root.map(|h| h.to_hex()),
76                "metadata_included": metadata
77            });
78            println!("{}", serde_json::to_string_pretty(&output_info)?);
79        } else {
80            println!("{} Content written to: {}", 
81                "✓".green().bold(), 
82                output_path.display().to_string().bright_white());
83            
84            if metadata {
85                println!("  {} Size: {} bytes", "→".cyan(), content.len());
86                if verify {
87                    println!("  {} Content verified", "✓".green());
88                }
89            }
90        }
91    } else {
92        // Stream to stdout (default behavior)
93        if json {
94            // JSON metadata to stderr, content to stdout
95            let output_info = serde_json::json!({
96                "action": "content_streamed",
97                "path": path,
98                "size": content.len(),
99                "verified": verify,
100                "at_root": at_root.map(|h| h.to_hex()),
101                "metadata_included": metadata
102            });
103            eprintln!("{}", serde_json::to_string_pretty(&output_info)?);
104        }
105        
106        // Always stream content to stdout
107        std::io::stdout().write_all(&content)?;
108        
109        if metadata && !json {
110            eprintln!("{} Size: {} bytes", "→".cyan(), content.len());
111            if verify {
112                eprintln!("  {} Content verified", "✓".green());
113            }
114        }
115    }
116
117    Ok(())
118}
