3use crate::core::error::{DigstoreError, Result};
4use serde::{Deserialize, Serialize};
5use std::time::{SystemTime, UNIX_EPOCH};

7const GITHUB_RELEASES_URL: &str =
8    "https://api.github.com/repos/DIG-Network/digstore/releases/latest";
9const UPDATE_CHECK_INTERVAL: u64 = 24 * 60 * 60; // 24 hours in seconds

11/// Information about available updates
12#[derive(Debug, Clone, Serialize, Deserialize)]
13pub struct UpdateInfo {
14    pub current_version: String,
15    pub latest_version: String,
16    pub update_available: bool,
17    pub download_url: Option<String>,
18    pub release_notes: Option<String>,
19}

21/// GitHub release API response
22#[derive(Debug, Deserialize)]
23struct GitHubRelease {
24    tag_name: String,
25    name: String,
26    body: Option<String>,
27    assets: Vec<GitHubAsset>,
28}

30/// GitHub release asset
31#[derive(Debug, Deserialize)]
32pub struct GitHubAsset {

37/// Check for available updates
38pub fn check_for_updates() -> Result<UpdateInfo> {
39    let current_version = env!("CARGO_PKG_VERSION").to_string();
40
41    // Check if we should skip update check (too recent)
42    if !should_check_for_updates()? {
43        return Ok(UpdateInfo {
44            current_version: current_version.clone(),
45            latest_version: current_version,
46            update_available: false,
47            download_url: None,
48            release_notes: None,
49        });
50    }
51
52    // Fetch latest release info from GitHub
53    let latest_release = fetch_latest_release()?;
54    let latest_version = latest_release.tag_name.trim_start_matches('v').to_string();
55
56    // Compare versions
57    let update_available = is_newer_version(&latest_version, &current_version)?;
58
59    // Find appropriate download URL for current platform
60    let download_url = if update_available {
61        find_platform_download_url(&latest_release.assets)
62    } else {
63        None
64    };
65
66    // Update last check timestamp
67    update_last_check_timestamp()?;
68
69    Ok(UpdateInfo {
70        current_version,
71        latest_version,
72        update_available,
73        download_url,
74        release_notes: latest_release.body,
75    })
76}

78/// Check if we should perform an update check (respects interval)
79fn should_check_for_updates() -> Result<bool> {
80    let config_path = get_update_config_path()?;
81
82    if !config_path.exists() {
83        return Ok(true); // First time, always check
84    }
85
86    match std::fs::read_to_string(&config_path) {
87        Ok(content) => {
88            if let Ok(last_check) = content.trim().parse::<u64>() {
89                let now = SystemTime::now()
90                    .duration_since(UNIX_EPOCH)
91                    .unwrap_or_default()
92                    .as_secs();
93
94                Ok(now - last_check > UPDATE_CHECK_INTERVAL)
95            } else {
96                Ok(true) // Invalid timestamp, check anyway
97            }
98        },
99        Err(_) => Ok(true), // Can't read file, check anyway
100    }
101}

103/// Update the last check timestamp
104fn update_last_check_timestamp() -> Result<()> {
105    let config_path = get_update_config_path()?;
106
107    if let Some(parent) = config_path.parent() {
108        std::fs::create_dir_all(parent)?;
109    }
110
111    let now = SystemTime::now()
112        .duration_since(UNIX_EPOCH)
113        .unwrap_or_default()
114        .as_secs();
115
116    std::fs::write(&config_path, now.to_string())?;
117    Ok(())
118}

120/// Get path to update config file
121fn get_update_config_path() -> Result<std::path::PathBuf> {
122    use directories::UserDirs;
123
124    let user_dirs = UserDirs::new().ok_or(DigstoreError::HomeDirectoryNotFound)?;
125    let dig_dir = user_dirs.home_dir().join(".dig");
126    Ok(dig_dir.join("last_update_check"))
127}

129/// Fetch latest release from GitHub API
130fn fetch_latest_release() -> Result<GitHubRelease> {
131    let client = reqwest::blocking::Client::builder()
132        .user_agent("digstore-cli")
133        .timeout(std::time::Duration::from_secs(10))
134        .build()
135        .map_err(|e| DigstoreError::NetworkError {
136            reason: format!("Failed to create HTTP client: {}", e),
137        })?;
138
139    let response =
140        client
141            .get(GITHUB_RELEASES_URL)
142            .send()
143            .map_err(|e| DigstoreError::NetworkError {
144                reason: format!("Failed to fetch release info: {}", e),
145            })?;
146
147    if !response.status().is_success() {
148        return Err(DigstoreError::NetworkError {
149            reason: format!("GitHub API returned status: {}", response.status()),
150        });
151    }
152
153    let release: GitHubRelease = response.json().map_err(|e| DigstoreError::NetworkError {
154        reason: format!("Failed to parse release info: {}", e),
155    })?;
156
157    Ok(release)
158}

160/// Compare two version strings (semver-like)
161fn is_newer_version(latest: &str, current: &str) -> Result<bool> {
162    // Simple version comparison - parse as semver-like
163    let parse_version = |v: &str| -> Result<(u32, u32, u32)> {
164        let parts: Vec<&str> = v.split('.').collect();
165        if parts.len() != 3 {
166            return Err(DigstoreError::ConfigurationError {
167                reason: format!("Invalid version format: {}", v),
168            });
169        }
170
171        let major = parts[0]
172            .parse::<u32>()
173            .map_err(|_| DigstoreError::ConfigurationError {
174                reason: format!("Invalid major version: {}", parts[0]),
175            })?;
176        let minor = parts[1]
177            .parse::<u32>()
178            .map_err(|_| DigstoreError::ConfigurationError {
179                reason: format!("Invalid minor version: {}", parts[1]),
180            })?;
181        let patch = parts[2]
182            .parse::<u32>()
183            .map_err(|_| DigstoreError::ConfigurationError {
184                reason: format!("Invalid patch version: {}", parts[2]),
185            })?;
186
187        Ok((major, minor, patch))
188    };
189
190    let latest_version = parse_version(latest)?;
191    let current_version = parse_version(current)?;
192
193    Ok(latest_version > current_version)
194}

196/// Find download URL for current platform
197fn find_platform_download_url(assets: &[GitHubAsset]) -> Option<String> {
198    let platform_patterns = if cfg!(target_os = "windows") {
199        vec!["windows-x64.msi", "windows.msi"]
200    } else if cfg!(target_os = "macos") {
201        vec!["macos.dmg", "darwin.dmg"]
202    } else if cfg!(target_os = "linux") {
203        vec!["linux-x86_64.AppImage", "linux.AppImage", "amd64.deb"]
204    } else {
205        vec![]
206    };
207
208    for pattern in platform_patterns {
209        for asset in assets {
210            if asset.name.contains(pattern) {
211                return Some(asset.browser_download_url.clone());
212            }
213        }
214    }
215
216    None
217}
